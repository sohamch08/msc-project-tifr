\section{Character Sums with Polynomials}\label{sec:character-sum-poly}
Let $\chi\in\Xq$ be a non-trivial additive character of $\bbF_q$ and let $f\in\bbF_q[X]$ be a non-constant polynomial. In this section we will look sums of the form $$\sum_{\alpha\in\bbF_q}\chi(f(\alpha))$$Such sums are called \emph{Weil Sums.}
\begin{Definition}{Weil Sum}{}
  \index{Weil sum}%
  Let $\chi\in\Xq$ be a non-trivial additive character of $\bbF_q$ and let $f\in\bbF_q[X]$ be a non-constant polynomial. The \emph{Weil sum} associated to $\chi$ and $f$ is
  $$W(\chi,f)\coloneqq\sum_{\alpha\in\bbF_q}\chi(f(\alpha)).$$
  More generally, for a multiplicative character $\psi\in\Mq$ and a rational function $f\in\bbF_q(X)$, the \emph{Weil sum} is
  $$W(\psi,\chi,f)\coloneqq\sum_{\substack{\alpha\in\bbF_q\\ \alpha\text{ not a pole of }f}}\psi(\alpha)\,\chi(f(\alpha)).$$
\end{Definition}Weil sums are extremely difficult to evaluate. One usually has to be satisfied with estimates for the absolute value of the sum. In certain cases it is easier to find the character sums exactly. 

\subsection{Additive Character Sums}

The first family of Weil sums we evaluate exactly are those of the form $\chi(ac^n+b)$, where the polynomial is a shifted monomial. The key is to factor the sum over $c$ using multiplicative characters of order $d=\gcd(n,q-1)$.

\begin{Theorem}{Weil Sum of a Shifted Monomial}{thm:weil-shifted-monomial}
  Let $\chi\in\Xq$ be a non-trivial additive character of $\bbF_q$, $n\in\bbN$, and let $\lm\in\Mq$ be a multiplicative character of order $d=\gcd(n,q-1)$. Then for any $a,b\in\bbF_q$ with $a\neq 0$,
  $$\sum_{\gm\in\bbF_q}\chi(a\cdot\gm^n+b)=\chi(b)\sum_{j=1}^{d-1}\ov{\lm}^j(a)\,G(\lm^j,\chi).$$
\end{Theorem}
\begin{proof}
  Let $\tau\in\Xq$ be a non-trivial additive character of $\bbF_q$ defined by $\tau(\gm)=\chi(a\cdot \gm)$ for all $\gm\in\bbF_q$. Then we get $$\sum_{\gm\in\bbF_q}\chi(a\cdot \gm^n+b)=\chi(b)\sum_{\gm\in\bbF_q}\chi(a\cdot \gm^n)=\chi(b)\sum_{\gm\in\bbF_q}\tau(\gm^n)$$
  Now by \obsref[(ii)]{obs:fourier-coeff-gaussian-sum} we get $\tau(\gm^n)=\frac1{q-1}\sum_{\psi\in\Mq}G(\ov{\psi},\tau)\cdot \psi(\gm^n)$ for all $\gm\in\bbF_q$. Therefore we get \begin{align*}
    \sum_{\gm\in\bbF_q}\tau(\gm^n) & = \tau(0)+\sum_{\gm\in\bbF_q^*}\frac1{q-1}\sum_{\psi\in\Mq}G(\ov{\psi},\tau)\cdot \psi(\gm^n)\\ 
    & = \tau(0)+\frac1{q-1}\sum_{\psi\in\Mq}G(\ov{\psi},\tau)\sum_{\gm\in\bbF_q^*}\psi^n(\gm)
  \end{align*}
  Now if $\psi^n$ is trivial then $\sum_{\gm\in\bbF_q^*}\psi^n(\gm)=q-1$ and if $\psi^n$ is non-trivial then $\sum_{\psi\in\Mq}\psi^n(\gm)=0$ by \MultCharPropsthm[(i)]. Now $\psi^n$ is trivial if and only if $\ord(\psi)\mid d$. Since $\ord(\ov{lm})=d$, the characters $\psi$ with order dividing $d$ are exactly $\lm^j$ for all $0\leq j\leq d-1$. Therefore $$\sum_{\gm\in\bbF_q}\tau(\gm^n)=1+\sum_{j=0}^{d-1}G(\lm^j,\tau) = \sum_{j=1}^{d-1}G(\lm^j,\tau)=\sum_{j=1}^{d-1}\ov{\lm}^j(a)\cdot G(\lm^j,\chi)$$Hence we have the theorem. 
\end{proof}

From \GSpropthm we have the absolute value of $G(\psi,\chi)$ to be $q^{\nicefrac12}$ if $\chi,\psi$ are non-trivial additive and multiplicative characters respectively. So we get the absolute value of the above character sum to be at most $(d-1)q^{\nicefrac12}$. Therefore as an immediate consequence we get the following bound on the absolute value of such sums.

\begin{Theorem}{Bound on Weil Sum of a Shifted Monomial}{thm:weil-shifted-monomial-bound}
  Let $\chi\in\Xq$ be a non-trivial additive character of $\bbF_q$, $n\in\bbN$, and $d=\gcd(n,q-1)$. Then for any $a,b\in\bbF_q$ with $a\neq 0$,
  $$\lt|\sum_{c\in\bbF_q}\chi(ac^n+b)\rt|\leq (d-1)\sqrt{q}.$$
\end{Theorem}

So if $\gcd(n,q-1)=1$ then the sum at right hand side is over no elements. So the sum evaluates to be zero. So we have the following corollary:
\begin{corolary}{Vanishing for Coprime Exponent}{cor:weil-vanishing-coprime}
  Let $\chi\in\Xq$ be a non-trivial additive character of $\bbF_q$ and $n\in\bbN$. If $\gcd(n,q-1)=1$ then for any $a,b\in\bbF_q$ with $a\neq 0$,
  $$\sum_{c\in\bbF_q}\chi(ac^n+b)=0.$$
\end{corolary}

For quadratic polynomials in odd characteristic the Weil sum can be evaluated exactly in terms of a Gaussian sum and the quadratic character. 

\begin{Theorem}{Weil Sum of a Quadratic Polynomial}{thm:weil-quadratic-odd}
  Let $\chi\in\Xq$ be a non-trivial additive character of $\bbF_q$ with $q$ odd, and let $f(X)=a_2X^2+a_1X+a_0\in\bbF_q[X]$ with $a_2\neq 0$. Then
  $$\sum_{\gm\in\bbF_q}\chi(f(\gm))=\chi\lt(a_0-a_1^2(4a_2)^{-1}\rt)\,\eta(a_2)\,G(\eta,\chi),$$
  where $\eta\in\Mq$ is the quadratic character of $\bbF_q$.
\end{Theorem}
\begin{proof}
  Since $f$ is a quadratic polynomial we can express $f$ as in the form of \thmref{thm:weil-shifted-monomial} for degree $2$. $$f(X)=a_2X^2+a_1X+a_0=a_2\lt(X+a_1\cdot (2a_2)^{-1}\rt)^2+\lt(a_0-a_1^2(4a_2)^{-1}\rt)$$Let $\ov{\gm}=\gm+a_1(2a_2)^{-1}$ and $\beta=a_0-a_1^2(4a_2)^{-1}$. Then we have $$\sum_{\gm\in\bbF_q}\chi(f(\gm))=\sum_{\ov{\gm}\in\bbF_q}\chi(a_2\cdot \ov{\gm}^2+\beta)$$So by \thmref{thm:weil-shifted-monomial}$$\sum_{\gm\in\bbF_q}\chi(f(\gm))=\chi(\beta)\cdot \eta(a_2)\cdot G(\eta,\chi)$$Hence we have the result. 
\end{proof}
For polynomials whose monomials are all \emph{affine $p$-polynomials} i.e. powers of $p = \Char(\bbF_q)$, the sum collapses to either $0$ or a full power of $q$ depending on a linear condition on the parameter $b$.

\begin{Theorem}{Weil Sum of an Affine $p$-Polynomial}{thm:weil-p-polynomial}
  Let $p=\Char(\bbF_q)$ and let
  $$f(X)=a_rX^{p^r}+a_{r-1}X^{p^{r-1}}+\cdots+a_1X^p+a_0X+a\in\bbF_q[X]$$
  be an affine $p$-polynomial over $\bbF_q$. Let $\chi_b\in\Xq$, $b\in\bbF_q^*$, be a non-trivial additive character as defined in \AddCharDefthm. Then
  $$\sum_{\gm\in\bbF_q}\chi_b(f(\gm))=\begin{cases}
    \chi_b(a)\cdot q & \text{if }ba_r+b^{\,p}a_{r-1}+\cdots+b^{p^{r-1}}a_1^{p^{r-1}}+b^{p^r}a_0^{p^r}=0,\\
    0 & \text{otherwise}
  \end{cases}$$
\end{Theorem}
\begin{proof}
  We can bring out the $\chi_b(a)$ factor from $\chi_b(f(\gm))$ as $a$ is the constant term. Consider the following polynomial $F(X)=b\cdot f(X)$. So we have $$\sum_{\gm\in\bbF_q}\chi_b(f(\gm))=\chi_b(a)\sum_{\gm\in\bbF_q}\chi_1(F(\gm))$$where $\chi_1$ is the canonical additive character of $\bbF_q$. Let $\tau$ be the additive character defined as $\tau(\gm)=\chi_1(F(\gm))$ for all $\gm\in\bbF_q$. Then by \AddCharPropsthm[(i)] we have $$\sum_{\gm}\tau(\gm)=\begin{cases}
    q & \text{if $\tau$ is trivial}\\ 0 & \text{otherwise}
  \end{cases}$$So now it suffices to characterize those affine $p$-polynomials $F(X)$ for which $\tau$ is trivial. Let $q=p^s$ and let $\tr$ is the absolute trace function from $\bbF_q$ to $\bbF_p$. Then $\tau$ is trivial if and only if $\tr(F(\gm))=0$ for all $\gm\in\bbF_q$. Therefore we get $$\tau\text{ is trivial}\iff \sum_{j=0}^{s-1}F^{p^j}(X)\equiv 0\bmod{\lt\la X^q-X\rt\ra}$$ Now we have $$\sum_{j=0}^{s-1}F^{p^j}(X)=\sum_{j=0}^{s-1}\lt(\sum_{i=0}^{r}b\cdot a_i\cdot X^{p^i}\rt)^{p^j}=\sum_{j=0}^{s-1}\sum_{i=0}^{r}b^{p^j}\cdot a_i^{p^j}\cdot X^{p^{i+j}}\equiv \sum_{k=0}^{s-1}\lt(\sum_{i=0}^rb^{p^{k-i}}a_i^{p^{k-i}}\rt)X^{p^k}\bmod{\lt\la X^q-X\rt\ra}$$Therefore we get $$\sum_{j=0}^{s-1}F^{p^j}(X)\equiv 0\bmod{\lt\la X^q-X\rt\ra}\iff \sum_{i=0}^rb^{p^{k-i}}a_i^{p^{k-i}}=0,\ \forall\ 0\leq k\leq s-1$$

  Now for all $j\in\{0,\dots, s-1\}$ we have $$\sum_{i=0}^rb^{p^{k-i}}a_i^{p^{k-i}}=0\iff \lt(\sum_{i=0}^rb^{p^{k-i}}a_i^{p^{k-i}}\rt)^{p^{r-k}}=0\iff \sum_{i=0}^rb^{p^{r-i}}a_i^{p^{r-i}}=0$$So we get the Theorem. 
\end{proof}
In the above theorem the case when $q$ is even is also studies while in \thmref{thm:weil-quadratic-odd} its not been studied. Now for the above theorem if $f$ is a quadratic polynomial we can extend \thmref{thm:weil-quadratic-odd} for $q$ is even case and we get the following corollary: 
\begin{corolary}{Weil Sum of Quadratic Polynomial in Even Characteristic}{cor:weil-quadratic-even}
  Let $f(X)=a_2X^2+a_1X+a_0\in\bbF_q[X]$ with $q$ even, and let $\chi_b\in\Xq$, $b\in\bbF_q^*$, be as defined in \thmref{thm:weil-p-polynomial}. Then
  $$\sum_{c\in\bbF_q}\chi_b(f(c))=\begin{cases}
    \chi_b(a_0)\cdot q & \text{if }a_2=ba_1^2,\\
    0 & \text{otherwise.}
  \end{cases}$$
\end{corolary}

\subsection{Quadratic Character Sums}

When the character is the quadratic character $\eta$, the sums $\sum_c\eta(f(c))$ can be evaluated by completing the square (for quadratic $f$) or by the continued fraction algorithm (for polynomials with no roots in $\bbF_q$).
\begin{Theorem}{Quadratic Character Sum of a Quadratic Polynomial}{thm:eta-quadratic-poly}
  Let $f(X)=a_2X^2+a_1X+a_0\in\bbF_q[X]$ with $q$ odd and $a_2\neq 0$, let $\Dl=a_1^2-4a_0a_2$ denote the discriminant of $f$, and let $\eta$ be the quadratic character of $\bbF_q$ (extended by $\eta(0)=0$). Then $$\sum_{\gm\in\bbF_q}\eta(f(\gm))=\begin{cases}-\eta(a_2) & \text{if }\Dl\neq 0,\\ (q-1)\cdot \eta(a_2) & \text{if }\Dl=0.\end{cases}$$
\end{Theorem}
\begin{proof}
  Since $4a_2^2=(2a_2)^2$ we have $\eta(4a_2)=1$. So we multiply the sum by $\eta(4a_2)^2$ and the sum stays the same. So we get for any $\gm\in\bbF_q$ $$\eta(f(\gm))\eta(4a_2^2\cdot f(\gm))= \eta(a_2)\cdot \eta(4a_2^2\cdot \gm^2+4a_1a_2\cdot \gm+4a_2a_1)=\eta(a_2)\cdot \eta\lt( (2a_2\cdot \gm+a_1)^2-\Dl \rt)$$This gives $\sum_{\gm\in\bbF_q}\eta(f(\gm))=\eta(a_2)\sum_{\ov{\gm}\in\bbF_q}\eta(\ov{\gm}^2-\Dl)$ where for any $\gm\in\bbF_q$ define $\ov{\gm}=2a_2\cdot \gm+a_1$. Now if $\Dl=0$ then $\eta(\ov{\gm}^2)=1$ and hence $\sum_{\gm\in\bbF_q}\eta(f(\gm))=-\eta(a_2)$. 
  
  So suppose $\Dl\neq 0$. Let for any $\beta\in\bbF_q$, $Z(\beta)$ is the number of solutions of $\ov{\gm}\in\bbF_q$ such that $\ov{\gm}^2=\beta^2-\Dl$. Then $\sum_{\beta\in\bbF_q}Z(\beta)$ is the total number of ordered pairs $(\ov{\gm},\beta)\in\bbF_q^2$ such that $\ov{\gm}^2-\beta^2=\Dl$. Now $$\ov{\gm}^2-\beta^2=\Dl\iff (\ov{\gm}+\beta)(\ov{\gm}-\beta)=\Dl$$Since $\Dl\neq 0$, both $\ov{\gm}+\beta$ and $\ov{\gm}-\beta$ are nonzero. Take $u=\ov{\gm}+\beta$ and $v=\ov{\gm}-\beta$. Then there is a one-one correspondence for each $(\ov{\gm},\beta)\in\bbF_q^2$ such that $\ov{\gm}^2-\beta^2=\Dl$ and each $(u,v)\in\bbF_q^2$ such that $u\cdot v=\Dl$. For any $u\in\bbF_q^*$, we automatically get the value $v$ hence number of such $(u,v)\in\bbF_q^2$ such that $u\cdot v=\Dl$ is $q-1$. Therefore $\sum_{\beta\in\bbF_q}Z(\beta)=q-1$. So we have the result. 
\end{proof}

\subsection{Weil Sums of Continued Fractions}

There is a remarkable connection between Weil sums for the quadratic character $\eta$ and the \emph{continued fraction algorithm} for rational functions over $\bbF_q$. We collect the necessary algebraic facts about continued fractions of rational functions in Appendix~\ref{sec:cont-frac-fq}; here we set up the specific expansion needed for the application to character sums and state the main results.

For $q$ odd, put $G(X)=X^q-X$, let $f\in\bbF_q[X]$ be a polynomial of positive degree with no roots in $\bbF_q$, and set $F(X)=f(X)^{(q-1)/2}$. Consider the continued fraction expansions
$$\frac{F(X)-1}{G(X)}=[A_0,A_1,\dots,A_s]\qquad\text{and}\qquad\frac{F(X)+1}{G(X)}=[a_0,a_1,\dots,a_t]$$
with $P_i,Q_i$ and $p_i,q_i$ the corresponding numerator/denominator sequences from Appendix~\ref{sec:cont-frac-fq}. It is clear that $A_0=a_0$. Define $n_f$ to be the largest integer $m$ such that $A_i=a_i$ for $i=0,1,\dots,m$.

\begin{lemma}{}{lem:cf-nf-cases}
  For $q$ odd, put $G(X)=X^q-X$, let $f\in\bbF_q[X]$ be a polynomial of positive degree with no roots in $\bbF_q$, and set $F(X)=f(X)^{(q-1)/2}$. Let $$\frac{F(X)-1}{G(X)}=[A_0,A_1,\dots,A_s]\qquad\text{and}\qquad\frac{F(X)+1}{G(X)}=[a_0,a_1,\dots,a_t]$$Then either $n_f=s=t-1$ or $n_f=t=s-1$.
\end{lemma}
\begin{proof}
  Let $P_i,Q_i$ are defined as above from $[A_0,\dots, A_s]$ and $p_i,q_i$ are defined from $[a_0,\dots,a_t]$. Then using \corref{cor:piqi-is-expression} and \corref{cor:piqi-coprime} we have $$\frac{P_s}{Q_s}=\frac{F(X)-1}{G(X)}\qquad\text{and}\qquad\frac{p_t}{q_t}=\frac{F(X)+1}{G(X)}$$So we get 
  \begin{equation}\label{eq:Qs-gx-qt}
    Q_s(X)=\frac{\alpha_1\cdot G(X)}{\gcd(F(X)-1,G(X))},\quad q_t(X)=\frac{\alpha_2\cdot G(X)}{\gcd(F(X)+1,G(X))}\quad\text{for some }\alpha_1,\alpha_2\in\bbF_q^*
  \end{equation}
  Since $f$ has no roots in $\bbF_q$, for all $\alpha\in\bbF_q$, $f^{q-1}(\alpha)=1$. So $G(X)\mid f^{q-1}(X)-1$ i.e. $G(X)\mid (F(X)-1)(F(X)+1)$. So $$\gcd(G(X),F(X)-1)\cdot\gcd(G(X),F(X)+1)=G(X)$$Therefore $Q_s(X)\cdot q_t(X)=\alpha \cdot G(X)$ with $\alpha\in\bbF_q^*$. Let $n=n_f<s$ and $n<t$. Therefore by \lemref{lem:cf-r0r1-rep} we get $$\frac{F(X)-1}{G(X)}=\frac{P_{n+1}+\beta_{n+1}\cdot P_n}{Q_{n+1}+\beta_{n+1}\cdot Q_n},\quad\text{ and }\quad\frac{F(X)+1}{G(X)}=\frac{p_{n+1}+\gm_{n+1}\cdot p_n}{q_{n+1}+\gm_{n+1}\cdot q_n}$$with $\beta_{n+1},\gm_{n+1}$ being rational functions with negative degree. Therefore 
  \begin{equation}\label{eq:gx-hx}
    \frac{2}{G(X)}=\frac{F(X)+1}{G(X)}-\frac{F(X)-1}{G(X)}=\frac{H(X)}{(Q_{n+1}+\beta_{n+1}\cdot Q_n)(q_{n+1}+\gm_{n+1}\cdot q_n)}
  \end{equation}
  where \begin{multline}\label{eq:cont-frac-2GX}
    H  = \underbrace{p_{n+1}\cdot Q_{n+1}-P_{n+1}\cdot q_{n+1}}_{H_0}\;+\;\beta_{n+1}(\underbrace{p_{n+1}\cdot Q_n-P_n\cdot q_{n+1}}_{H_1})  
    \;+\; \gm_{n+1}(\underbrace{p_n\cdot Q_{n+1}-P_{n+1}\cdot q_n}_{H_2})\\ 
    \;+\;\beta_{n+1}\cdot \gm_{n+1}(\underbrace{p_n\cdot Q_n-P_n\cdot q_n}_{H_3}) 
  \end{multline}
  Now by definition of $n$ we have $P_i=p_i$ and $Q_i=q_i$ for all $-1\leq i\leq n$. By \lemref{lem:cf-determinant} we have $H_1=(-1)^n$ and $H_2=(-1)^{n+1}$ and $H_3=0$. So now we have to calculate $H_0$. 
  \begin{align*}
    H_0 & = p_{n+1}\cdot Q_{n+1}-P_{n+1}\cdot q_{n+1}\\ 
    & = (A_{n+1}\cdot p_n+p_{n-1})(A_{n+1}\cdot Q_n+Q_{n-1})-(A_{n+1}\cdot P_n+P_{n-1})(a_{n+1}\cdot q_n+q_{n-1})\\ 
    & = a_{n+1}(p_n\cdot Q_{n-1}-P_n\cdot q_{n-1}) +A_{n+1}(p_{n-1}\cdot Q_n-q_{n-1}\cdot P_n)\\ 
    &= (-1)^{n-1}a_n-(-1)^{n-1}A_{n+1}\\ 
    & = (-1)^n(A_{n+1}-a_{n+1})
  \end{align*}
  Therefore $H=(-1)^n(A_{n+1}-a_{n+1}+\beta_{n+1}-\gm_{n+1})$. Now by definition of $n$, $A_{n+1}\neq a_{n+1}$. So $\deg(H)\geq 0$. Therefore from the equation \eqref{eq:gx-hx} we get $G(X)\cdot H(X)=2(Q_{n+1}+\beta_{n+1}\cdot Q_n)(q_{n+1}+\gm_{n+1}\cdot q_n)$. By the definition of $G$ we have $\deg(G)+\deg(H)\geq q$. On the other hand, $\deg(G)+\deg(H)=\deg(Q_{n+1})+\deg(q_{n+1})\leq \deg(Q_s)+\deg(q_t)\leq q$. Therefore we have $\deg(H)=0$. So $\deg(A_{n+1})=\deg(a_{n+1})$. And hence $\deg(Q_{n+1})=\deg(q_{n+1})=\nicefrac{q}2$. But we have $q$ to be odd. Hence contradiction \ctr So we have either $n=s$ or $n=t$. Suppose $n=s$. Then $t>s$ and therefore we can write $$\frac{F(X)+1}{G(X)}=\frac{p_{s+1}+\gm_{s+1}\cdot p_s}{q_{s+1}+\gm_{s+1}\cdot q_s}$$Therefore we get new equality for \eqref{eq:gx-hx} 
  \begin{align*}
    \frac{2}{G(X)} & = \frac{F(X)+1}{G(X)}-\frac{F(X)-1}{G(X)}\\ 
    & = \frac{p_{s+1}+\gm_{s+1}\cdot p_s}{q_{s+1}+\gm_{s+1}\cdot q_s}-\frac{P_s}{Q_s}\\ 
    & = \frac{Q_s\cdot p_{s+1}-P_s\cdot q_{s+1}+\gm_{s+1}(Q_s\cdot p_s-P_s\cdot q_s)}{Q_s(q_{s+1}+\gm_{s+1}\cdot q_s)}\\ 
    & = \frac{(-1)^s}{Q_s(q_{s+1}+\gm_{s+1}\cdot q_s)}\byl{cf-determinant}
  \end{align*}
  So we have $(-1)^s\cdot G=2\cdot Q_s(q_{s+1}+\gm_{s+1}\cdot q_s)$. Again we will compare the degrees. And since $\deg(G)=q$ we have $\deg(Q_s)+\deg_{q_{s+1}}=q$. On the other hand we have $\deg(Q_s)+\deg_{q_{s+1}}\leq \deg(Q_s)+\deg(q_t)=q$. Therefore $\deg(q_{s+1})=\deg(q_t)$. Since degrees of $q_i$ increases strictly we have $t=s+1$. Similarly, if $n=t$ then we get $t+1=s$.
\end{proof}
So now we can derive the Weil sum of quadratic character for any polynomial $f$ which has no rooots in $\bbF_q$ where $q$ is odd.

\begin{Theorem}{Weil Sum via Continued Fractions}{thm:weil-cf}
  Let $\eta\in\Mq$ be the quadratic character of $\bbF_q$, $q$ odd, and let $f\in\bbF_q[X]$ be a polynomial of positive degree with no roots in $\bbF_q$. Then
  $$\sum_{\gm\in\bbF_q}\eta(f(\gm))=\begin{cases}
    \deg(a_s) & \text{if }n_f=s,\\
    -\deg(A_s) & \text{if }n_f=t,
  \end{cases}$$
  where $A_s$ and $a_s$ are obtained from the continued fraction expansions above. 
\end{Theorem}

\begin{proof}
  We define the following two sets $Z_1\coloneqq|\{\gm\in\bbF_q\colon \eta(f(\gm))=1\}|$ and $Z_{-1}\coloneqq |\{\gm\in\bbF_q\colon \eta(f(\gm))=-1\}|$. Then basically we have $\sum_{\gm\in\bbF_q}\eta(f(\gm))=Z_1-Z_{-1}$.  Since $\eta(f(\gm))=1$ if and only if $F(\gm)=f^{\nicefrac{(q-1)}{2}}(\gm)=1$ we have $Z_1=\deg\lt(\gcd(F(X)-1,X^q-X)\rt)$ and hence by \eqref{eq:Qs-gx-qt} we have  $Z_1=q-\deg(Q_s)$. 
  
  Similarly $\eta(f(\gm))=-1$ if and only if $F(\gm)=-1$. Hence $Z_{-1}=\deg\lt(\gcd(F(X)+1,X^q-X)\rt)=q-\deg(q_t)$. So if $n_f=t=s-1$ then $q_t=Q_{s-1}$. And therefore $q_t=Q_{s-1}$. So $Z_1-Z_{-1}=-\deg(Q_s)+\deg(Q_{s-1})=\deg(A_s)$. In case of $n_f=s=t-1$ we get $Z_1-Z_{-1}=\deg(q_t)-\deg(q_{t-1})=\deg(a_t)$. So the result follows.  
\end{proof}
\begin{note}
  The hypothesis that $f$ has no roots in $\bbF_q$ is essential. For example, take $f(X)=X$: then $\sum_{\gm\in\bbF_q}\eta(f(\gm))=0$ by \MultCharPropsthm[(i)], yet $\frac{F(X)-1}{X^q-X}=[0,X^{\nicefrac{(q+1)}{2}}+X]$ and $\frac{F(X)+1}{X^q-X}=[0,X^{\nicefrac{(q+1)}{2}}-X]$, giving $s=t=1$ and $\deg(A_s)=\deg(a_t)=\nicefrac{q+1}{2}\neq 0$. The continued fraction formula does not reproduce the correct value in this case.
\end{note} 

In the next section we turn to a statistical perspective and study the average number of solutions of polynomial equations over $\bbF_q$. We will show that the average value of $Z(f)$ over all polynomials $f$ of bounded degree is exactly $q^{n-1}$, and then compute the variance of the solution count around this mean.
