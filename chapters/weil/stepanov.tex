\section{Stepanov Method: \texorpdfstring{$Y^d-f(X)$}{Yd-f(X)}}
\begin{Theorem}{}{thm:stepanov}
  Suppose that $Y^d-f(X)\in\bbF_q[X,Y]$ is an absolutely irreducible polynomial where $\deg(f)=m$. Suppose $q>100dm^2$. Then $$|Z(Y^d=f(X))-q|\leq 4d^{\nicefrac32}q^{\nicefrac12}$$
\end{Theorem}

\begin{Theorem}{}{thm:yd-fx-irr}
  Suppose $Y^d-f(X)\in\bbF_q[X,Y]$. Then the following are equivalent:
  \begin{enumerate}[label=(\roman*)]
    \item $Y^d-f(X)$ is absolutely irreducible. 
    \item $Y^d-c\cdot f(X)$ is absolutely irreducible for any $c\in\bbF_q^*$. 
    \item If $f$ factors into $f(X)=a\prod_{i=1}^m(X-\alpha_i)^{k_i}$ over $\ov{\bbF_q}$ where $\alpha_i\in\ov{\bbF_q}$ for some positive integers $k_1,\dots, k_m$. Then $\gcd(d,k_1,\dots, k_n)=1$. 
  \end{enumerate}
\end{Theorem}

\begin{Theorem}{}{}
  Suppose $d\mid q-1$ and let $\psi\in\Mq^{(d)}$ is a non-trivial multiplicative character of exponent $d$. Let $f(X)\in\bbF_q[X]$ is a polynomial of degree $m$. Let $Y^d-f(X)$ is absolutely irreducible and $q>100dm^3$. Then we have $$\lt|\sum_{\alpha\in\bbF_q}\psi(f(\alpha))\rt|<5md^{\nicefrac{3}{2}}q^{\nicefrac12}$$
\end{Theorem}

\begin{Theorem}{}{}
  Suppose $d\mid q-1$ and let $\psi\in\Mq^{(d)}$ is a non-trivial multiplicative character with $\ord(\psi)=d>1$. Let $f(X)\in\bbF_q[X]$ is a polynomial of degree $m$ which is not a $d^{th}$ power and $q>100dm^3$. Then we have $$\lt|\sum_{\alpha\in\bbF_q}\psi(f(\alpha))\rt|<5md^{\nicefrac{3}{2}}q^{\nicefrac12}$$
\end{Theorem}
