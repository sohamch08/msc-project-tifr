\section{Stepanov Method: \texorpdfstring{$Y^d-f(X)$}{ Yd-f(X)}}\label{sec:stepanov}
A curve of the form $Y^d=f(X)$ over $\bbF_q$, where $f\in\bbF_q[X]$ and $d\geq 2$, is called a \emph{superelliptic curve}\index{superelliptic curve}. When $d=2$ it is a \emph{hyperelliptic curve}\index{hyperelliptic curve}. The main result of this section gives a sharp bound on the number of $\bbF_q$-rational points on such a curve, proved via an auxiliary polynomial that vanishes with high multiplicity on the solution set. The theorem we will be proving in this section is the following:
\begin{Theorem}{Stepanov's Theorem}{thm:stepanov}
  \index{Stepanov method}\index{Weil bound!hyperelliptic}%
  Suppose that $Y^d-f(X)\in\bbF_q[X,Y]$ is an absolutely irreducible polynomial where $\deg(f)=m$.  Suppose $q>100dm^2$.  Then $$|Z(Y^d=f(X))-q|\leq 4d^{\nicefrac32}q^{\nicefrac12}.$$
\end{Theorem}

\subsection{Absolute Irreducibility Criteria}

The hypothesis that $Y^d - f(X)$ is absolutely irreducible is central to \Stepanov. The following theorem gives a concrete criterion in terms of the factorization of $f$ over $\ov{\bbF_q}$.
\begin{Theorem}{Characterisation of Absolute Irreducibility of $Y^d-f(X)$}{thm:yd-fx-irr}
  \index{absolutely irreducible}%
  Suppose $Y^d-f(X)\in\bbF_q[X,Y]$. Then the following are equivalent:
  \begin{enumerate}[label=(\roman*)]
    \item $Y^d-f(X)$ is absolutely irreducible.
    \item $Y^d-c\cdot f(X)$ is absolutely irreducible for any $c\in\bbF_q^*$.
    \item If $f$ factors into $f(X)=a\prod_{i=1}^m(X-\alpha_i)^{k_i}$ over $\ov{\bbF_q}$ where $\alpha_i\in\ov{\bbF_q}$ for some positive integers $k_1,\dots, k_m$, then $r=\gcd(d,k_1,\dots, k_m)=1$.
  \end{enumerate}
\end{Theorem}
\begin{proof}We will prove this lemma in three steps:
  \paragraph*{(i)$\implies$(ii):} Suppose $Y^d-c\cdot f(X)$ is reducible over $\ov{\bbF_q}[X]$. Since $c\in\bbF_q^*$, we can write $\lt(\nicefrac{Y}{\sqrt[d]{c}}\rt)^d - f(X)$. Therefore, $\lt(\nicefrac{Y}{\sqrt[d]{c}}\rt)^d - f(X)$ is also reducible over $\ov{\bbF_q}[X]$. But we have $Y^d-f(X)$ irreducible. Hence, contradiction \ctr So $Y^d-c\cdot f(X)$ is absolutely irreducible.

  \paragraph*{(ii)$\implies$(iii):} Assume the contrary i.e, $r>1$. Then consider the polynomial $$g(X)=\prod_{i=1}^{m}(X-\alpha_i)^{\nicefrac{k_i}r}$$Then we have $$Y^d-\nicefrac1a f(X)=Y^d-g(X)^r=(Y^{\nicefrac{d}r}-g(X))(g(X)^{r-1}+\cdots g(X)+1)$$Therefore, $Y^d-\nicefrac1a f(X)$ is reducible over $\ov{\bbF_q}[X]$. But we are given that $Y^d-c\cdot f(X)$ is absolutely irreducible for any $c\in\bbF_q^*$. Hence, contradiction \ctr We have $r=1$. 

  \paragraph*{(iii)$\implies$(i):} Let $E$ be the function field $\ov{\bbF_q}(X)$. Then we can consider $Y^d-f(X)$ as a polynomial of $E[Y]=\ov{\bbF_q}(X)[Y]$. Suppose $\gm\in \ovE$ be a root of $Y^d-f(X)$ and $\zeta_d$ be the $d^{th}$ root of unity. Then notice that $$Y^d-f(X)=(Y-\gm)(Y-\zeta_d\cdot \gm)\cdots (Y-\zeta_d^{d-1}\cdot \gm)$$Therefore, $\zeta_d^i\cdot \gm$ for all $0\leq i<d$ are the roots of $Y^d-f(X)$ over $\ovE$. Now suppose for the sake of contradiction assume $Y^d-f(X)$ is reducible over $\ov{\bbF_q}[X]$. Then there exists a $t\in[d]$ such that $$\lt(Y-\zeta_d^{i_1}\cdot \gm\rt)\cdots \lt(Y-\zeta_d^{i_t}\cdot \gm\rt)\in \ov{\bbF_q}[X,Y]$$ for some $0\leq i_1<\cdots <i_t\leq d-1$. Now the constant term of this polynomial is $\gm^t\prod_{j=1}^t\zeta_d^{i_j}$. Therefore, $\gm^t\in\ov{\bbF_q}[X]$. Let $l\in\bbZ_0$ be the smallest non-negative integer such that $\gm^l\in\ov{\bbF_q}[X]$. Then $l\mid d$. So consider the polynomial $h(X)=\gm^{l}$. 

  Now since $\gm$ is a root of $Y^d-f(X)$ we have $\gm^d=f(X)$. Therefore, $g^{\nicefrac{d}l}(X)=f(X)$. So take $k=\nicefrac{d}{l}$. Then $k\mid k_i$ for all $i\in[n]$. Therefore, we obtained $k>1$ such that $k\mid \gcd(d,k_1,\dots, k_m)$. Hence, contradiction \ctr So $Y^d-f(X)$ is absolutely irreducible. 
\end{proof}

The most useful special case of \YdFxIrrthm is when $\gcd(\deg f, d)=1$, which is the criterion used in the first proof of \Stepanov.
\begin{corolary}{Absolute Irreducibility from Coprime Degree}{cor:yd-fx-irr-coprime}
  Suppose $f\in\bbF_q[X]$ with $\deg(f)=m$. Then $Y^d-f(X)$ is absolutely irreducible if $\gcd(m,d)=1$.
\end{corolary}

\subsection{Proof of Stepanov's Theorem}
The proof reduces the solution count $Z(Y^d=f(X))$ to the count of $\alpha\in\bbF_q$ for which $f(\alpha)$ is a perfect $d$-th power; the following lemma makes this reduction explicit.
\begin{lemma}{Solution Count for $Y^d=f(X)$}{lem:stepanov-solution-count}
  Let $d\mid q-1$ and $f\in \bbF_q[X]$ be a polynomial. Let $F(X)=f^{\nicefrac{(q-1)}{d'}}(X)$ where $d'=\gcd(q-1,d)$. Then the number of solutions of $Y^d=f(X)$ in $\bbF_q^2$ is given by 
  $$Z(Y^d=f(X))=Z_0+d'\cdot Z_1$$ where $Z_0=|\{\alpha\in\bbF_q\colon f(\alpha)=0\}|$ and $Z_1=|\{\alpha\in\bbF_q\colon F(\alpha)=1\}|$.

  Furthermore, if $Z_2=|\{\alpha\in\bbF_q\colon F^{d'-1}(\alpha)+\cdots +F(\alpha)+1=0\}|$ then $Z_0+Z_1+Z_2=q$.
\end{lemma}
\begin{proof}
  Now for any solution $(X,Y)=(\alpha,\beta)$ such that $\beta^d=f(\alpha)$ either $\beta-0$ or $\beta\neq 0$. So we consider these two cases separately.
  \paragraph*{Case I: $\beta=0$} In this case the notice that the set of all $\alpha\in\bbF_q$ such that $f(\alpha)=0$ is exactly the set of solutions such that $\beta=0$. Therefore, the number of solutions for which $\beta=0$ is $Z_0$. 
  \paragraph*{Case II: $\beta\neq 0$} If for any solution $(\alpha,\beta)$ if $\beta\neq 0$ then $f(\alpha)\in {\bbF_q^*}^{(d)}$. Therefore, for all such $\alpha$'s we have $$f^{\nicefrac{(q-1)}{d'}}(\alpha)=F(\alpha)=1,\quad \text{where }d'=\gcd(q-1,d).$$ Now if $Z_1$ is the number of $\alpha\in\bbF_q$ such that $f(X)^{\nicefrac{(q-1)}{d'}})(\alpha)=1$. This is exactly the set $Z_1$. Now for each such $\alpha\in\bbF_q^*$ there are $d'$ solutions for $\beta$ in $\bbF_q$ such that $\beta^d=f(\alpha)$. Hence, there are $d'\cdot Z_1$ such solutions. 
  
  Thus, we get the total number of solutions is $Z_0+d'\cdot Z_1$. Hence, $Z(Y^d=f(X))=Z_0+d'\cdot Z_1$. So we have the first part
  
  Now for every $\alpha\in\bbF_q$, $f(\alpha)\in\bbF_q$. Therefore, $f^q(\alpha)=f(\alpha)$. Now we get the following factorization.
  $$H(X)=f^q(X)-f(X)=f(X)\cdot \lt(f^{\nicefrac{(q-1)}{d'}}-1\rt)\lt(F(X)^{d'-1}+\cdots + F(X)+1\rt).$$ So $\deg H=q$ and for all $\alpha\in\bbF_q$, $H(\alpha)=0$. Now if $H(\alpha)=0$ then either $f(\alpha)=0$ or $F(\alpha)=0$ or $F^{d-1}(\alpha)+\cdots +F(\alpha)+1=0$. The number of zeros in these three cases are $Z_0$, $Z_1$ and $Z_2$ respectively. Therefore, we get $Z_0+Z_1+Z_2=q$. 
\end{proof}
\subsubsection{Stepanov's Theorem with Restricted Conditions}
To prove \Stepanov we will assume $d\mid q-1$ and $\gcd(d,m)=1$, so $Y^d-f(X)$ is absolutely irreducible over $\bbF_q$. We also temporarily assume that $q=p$ or $q=p^2$ where $p$ is a prime.
The key linear-independence property below ensures that a vanishing relation among $h_0,\dots,h_{d-1}$ forces each coefficient $h_{ij}$ to vanish — it is what makes the polynomial method work.
\begin{lemma}{}{lem:stepanov-linear-indep}
  Let $m,d\in\bbN$ such that $\gcd(d,m)=1$. Suppose $h_0,h_1,\dots, h_{d-1}(X)\in\bbF_q[X]$ are polynomials of the type $$h_i(X)=\sum_{j=0}^th_{ij}(X)\cdot X^{qj},\quad \forall\ 0\leq i\leq d-1$$ where $h_{ij}\in\bbF_q[X]$ with $\deg(h_{ij})\leq \frac{q}{d}-m$. If $$h(X)=h_0(X)+h_1(X)\cdot F(X)+\cdots+h_{d-1}(X)\cdot F(X)^{d-1}\equiv 0$$ then for all $i\in\{0,1,\dots, d-1\}$ and $j\in\{0,\dots, t\}$ we have $h_{ij}\equiv 0$.
\end{lemma}
\begin{proof}
  A summand of $h(X)$ is of the form $h_{ij}(X)\cdot X^{qj}\cdot F(X)^i$. 

  \begin{want}
    We want to show that for any $i\in\{0,1,\dots, d-1\}$ and $j\in\{0,\dots, t\}$ the degree of term $q_{i,j}\coloneqq h_{ij}(X)\cdot X^{qj}\cdot F(X)^i$ are distinct. Then $h(X)\equiv 0$ directly implies $h_{ij}\equiv 0$ for all $i\in\{0,1,\dots, d-1\}$ and $j\in\{0,\dots, t\}$.
  \end{want}

  Now for any $i\in\{0,1,\dots, d-1\}$ and $j\in\{0,\dots, t\}$ we have $$\deg(q_{ij})=qj+i\;\frac{q-1}{d}m+\deg(h_{ij})=\frac{q}d(dj+im)-\frac{i}dm+\deg(h_{ij})$$Therefore, $$\frac{q}d(dj+im)-m<\deg(q_{ij})\leq \frac{q}d(dj+im)+\frac{q}d-m$$

  \begin{enough}
    For pairs $(i,j)\neq (i',j')$ we have $dj+im\neq dj'+i'm$
  \end{enough}

  Suppose there exists $(i,j)\neq (i',j')$ such that $dj+im=dj'+i'm$. Then we have $$im\equiv i'm\bmod d\implies i\equiv i'\bmod d$$where the implication comes from $\gcd(m,d)=1$. But $i,i'\in\{0,1,\dots, d-1\}$ so we have $i=i'$. So now we have $j=j'$. Hence, contradiction \ctr So the degrees of each summand in $h(X)$ are distinct. Therefore, $h_{ij}\equiv 0$ for all $i\in\{0,1,\dots, d-1\}$ and $j\in\{0,\dots, t\}$.
\end{proof}

The heart of the proof is the construction of a nonzero polynomial $Q$ of controlled degree that vanishes with high multiplicity on the zero set of $f$ and on the preimage of $g$ under $F$.
\begin{lemma}{Existence of Auxiliary Polynomial $Q$}{lem:stepanov-Q-exists}
  \index{auxiliary polynomial}%
  Let $1\leq r\leq d-1$ and let $g\in\bbF_q[X]$ be a polynomial of degree $r$. Let $M\in\bbN$ be a positive integer satisfying $M\geq m+1$ and $(M+3)^2\leq \nicefrac{2q}{d}$. Then there exists a non-zero polynomial $Q(X)\in\bbF_q[X]$ with $\deg(Q)\leq \nicefrac{r}{d}\cdot qM+4mq$ such that for every $\alpha\in\bbF_q$:
  \begin{enumerate}[label=(\roman*)]
    \item if $g(F(\alpha))=0$, then $Q(\alpha)=0$ with multiplicity at least $M$; and
    \item if $f(\alpha)=0$, then $Q(\alpha)=0$ with multiplicity at least $M$.
  \end{enumerate}
\end{lemma}
\begin{proof}
  We will construct $Q$ of the form $$Q(X)=f(X)^M\sum_{i=0}^{d-1}h_i(X)\cdot F(X)^i$$ where $h_i(X)=\sum_{j=0}^th_{ij}(X)\cdot X^{qj}$ with $\deg(h_{ij})\leq \frac{q}{d}-m$ for all $i\in\{0,1,\dots, d-1\}$ and $j\in\{0,\dots, t\}$ like \StepanovLinIndeplem. Take $t=\lt\lfloor (M+m+1)\nicefrac{r}{d}\rt\rfloor$. 
  \paragraph*{Computing the Hasse Derivatives.} \index{Hasse derivative}%
  We will calculate $H^{(n)}(Q)$ for $0\leq n\leq M-1$ for the multiplicity condition. Now for every term $f^M\cdot h_{ij}\cdot F^i$ we have $H^{(n)}(f^M\cdot h_{ij}\cdot F^i)=f^{M-n}h_{ij,n}\cdot F^i$ where 
  \begin{equation}\label{eq:hijn}
    \deg(h_{ij,n})\leq \deg(h_{ij})+n(k-1)\leq \frac{q}d-m+n(k-1)\leq \frac{q}{m}+n(k-1)-1
  \end{equation}
  \paragraph*{Condition for Multiplicity at zeros of $g(F(\alpha))$.} Now we can write $Q(X)=\tdQ(X,X^q)$ for some $\tdQ\in\bbF_q[X,Y]$ where $$\tdQ(X,Y)=f(X)^M\sum_{i=0}^{d-1}\sum_{j=0}^th_{ij}(X)\cdot Y^j\cdot F(X)^i$$Since $M\leq q$ we have $$H^{(n)}(Q(X))=f^{M-n}(X)\sum_{i=0}^{d-1}\sum_{j=0}^th_{ij,n}(X)\cdot F(X)^i\cdot X^{qj}.$$Let $g(X)=g_0+g_1\cdot X+\cdots g_r\cdot X^r$ where $g_i\in\bbF_q$ Then from these $g_i$'s we get $s_{ij}$'s such that for all $\alpha\in\bbF_q$ with $g(\alpha)=0$ we have $$\alpha^i=\sum_{i=0}^{r-1}s_{ij}\cdot \alpha^j$$ for all $0\leq i,j\leq r$. So if for any $\alpha\in\bbF_q$ we have $g(F(\alpha))=0$ we get $F^i(\alpha)=\sum_{i=0}^{r-1}s_{ij}\cdot F(\alpha)^j$. So now replacing with $s_{i,j}$'s we get for any $n\in\{0,1,\dots, M-1\}$ we get for any $\alpha\in\bbF_q$ such that $g(F(\alpha))=0$:
  \begin{align*}
    (H^{(n)}Q)(\alpha) & = f^{M-n}(\alpha)\sum_{i=0}^{d-1}\sum_{j=0}^th_{ij,n}(\alpha)\cdot F^i(\alpha)\cdot \alpha^{qj}\\ 
    & = f^{M-n}(\alpha)\sum_{i=0}^{d-1}\sum_{j=0}^th_{ij,n}(\alpha)\cdot F^i(\alpha)\cdot \alpha^{j} & [\alpha^q=\alpha\ \forall \ \alpha\in\bbF_q]\\ 
    & = f^{M-n}(\alpha)\sum_{i=0}^{d-1}\sum_{j=0}^t\sum_{l=0}^{r-1}h_{ij,n}(\alpha)\cdot F^i(\alpha)\cdot s_{j,l}\cdot \alpha^l \\ 
    & = f^{M-n}(\alpha)\sum_{l=0}^{r-1}q_{l,n}(\alpha)\cdot F^i(\alpha) & \lt[\text{where }q_{i,n}(X)=\sum_{i=0}^{d-1}\sum_{j=0}^th_{ij,n}(X)\cdot X^{j}\rt]
  \end{align*}
  \paragraph*{Counting Number of Equations.} From the degree bound of $h_{ij,n}$ in \eqref{eq:hijn} we have $\deg(q_{l,n})\leq \nicefrac{q}{d}+n(k-1)-1+t$. We want $(H^{(n)}Q)(\alpha)=0$ for all $n\in\{0,\dots, M-1\}$ for all $\alpha\in\bbF_q$ such that $g(F(\alpha))=0$. So it suffices to have the polynomials $q_{l,n}$ be identically zero. So if $s$ is the total number of equations $s$ then we have 
  \begin{multline}
    \label{eq:num-eq-yd-fx}
    s<\sum_{n=0}^{M-1}r\lt(\frac{q}{d}+n(k-1)+t\rt)=rM\lt(\frac{q}d+t\rt)+\frac12r(m-1)M(M-1)\leq \frac{rq}{d}M+rM\frac{r}{d}(M+m+1)+\frac12r(m-1)M^2\\ 
    <\frac{rq}{d}M+\frac12rM^2(m+1)+rM(m+1)
  \end{multline}
  Here the last inequality follows from the fact that $r<d$.
  \paragraph*{Counting Number of Variables.} Since we want a non-zero solution for the polynomials $h_i$ which satisfies the above equations, the coefficients of $h_{ij}$ are the variables. Let $v$ be the total number of possible coefficients of $h_{ij,n}$ for all $0\leq i\leq d-1$ and $0\leq j\leq t$. Then we have 
  \begin{equation}\label{eq:num-var-yd-fx}
    v\geq \lt(\frac{q}d-m\rt)d(t+1) \geq (q-md)\frac{r}{d}(M+m+1)=\frac{rq}{d}M+\frac{rq}{d}(m+1)-rm(M+k+1)\geq \frac{rq}{d}M+\frac{rq}{d}(m+1)-2rmM
  \end{equation}
  \paragraph*{Existence of non-zero solution.} So now if number of variables is more than the number of equations then we get a non-zero solution for the polynomials $h_{i,j}$ and henceforth for $h_i$. So we want $v>s$. From \eqref{eq:num-eq-yd-fx} and \eqref{eq:num-var-yd-fx} it suffices to have 
  \begin{align*}
    & \frac{rq}{d}M+\frac12rM^2(m+1)+rM(m+1)\leq \frac{rq}{d}M+\frac{rq}{d}(m+1)-2rmM\\ 
    \iff & \frac12rM^2(m+1)+rM(m+1)\leq\frac{rq}{d}(m+1)-2rmM\\ 
    \iff & \frac12M^2(m+1)+M(m+1)\leq \frac{q}d(m+1)-2mM\\ 
    \xLeftarrow{\hspace{12pt}}\; & \frac12M^2(m+1)+3M(m+1)\leq \frac{q}d(m+1)\\ 
    \iff & \frac12M^2+3M\leq \frac{q}d 
  \end{align*}
  The last inequality holds from the assumption that $(M+3)^2\leq \frac{2q}d$. So we get a non-zero solution for the polynomials $h_{ij}$ for all $0\leq i\leq d-1$ and $0\leq j\leq t$. Therefore, $Q$ is a non-zero polynomial otherwise \StepanovLinIndeplem will show every $h_{ij}$ is a zero polynomial. So $Q$ vanishes with multiplicity at least $M$ for all $\alpha\in\bbF_q$ such that $g(F(\alpha))=0$. 
  
  \paragraph*{Multiplicity at Zeros of $f$.} Since $Q$ has a factor $f^M(X)$, every $\alpha\in\bbF_q$ with $f(\alpha)=0$ is a zero of $Q$ with multiplicity at least $M$.
  \paragraph*{Degree Bound.} Since $\deg(h_{ij})\leq \frac{q}{d}-m$ we have $\deg(h_i)\leq \frac{q}{d}-m+tq$. Therefore,
  \begin{align*}
    \deg(Q) & \leq \underbrace{m M}_{\deg(f^M)}+\underbrace{\frac{q}{d}-m+qt}_{\deg(h_i)}+\underbrace{(d-1)\;\frac{q-1}{d}\;m}_{\deg(F^i)}\\ 
    & <mM+\frac{q}{d}+qm+\frac{qr}{d}(M+m+1)\\ 
    & <mq^{\nicefrac12}+\frac{r}{d}qM+q\lt(\frac1d+2m+1\rt)<\frac{r}{d}qM+4mq
  \end{align*}
  So we have the required degree bound for $Q$. Hence, we have the lemma.
\end{proof}

Combining \StepanovSolCountlem, \StepanovLinIndeplem, and \StepanovQExistslem, we now prove \Stepanov under the temporary assumptions $\gcd(d,m)=1$ and $q=p$ or $q=p^2$; these are removed in the following subsubsection.
\begin{Theorem}{Stepanov's Theorem under Simplifying Assumptions}{thm:stepanov-restricted}
  Let $d\mid q-1$ and $f\in \bbF_q[X]$ be a polynomial of degree $m$. Assume that \begin{enumerate}[label=(\roman*)]
    \item $\gcd(d,m)=1$, 
    \item $q\geq 100dm^2$
  \end{enumerate} then $$|Z(Y^d=f(X))-q|< 4md^{\nicefrac32}q^{\nicefrac12}.$$
\end{Theorem}
\begin{proof}
  For any $g\in\bbF_q[X]$ of degree $r$ in \StepanovQExistslem we constructed a polynomial $Q\in\bbF_q[X]$ where for all $\alpha\in\bbF_q$ such that $g(F(\alpha))=0$ or $f(\alpha)=0$, $\alpha$ is a zero of $Q$ with multiplicity at least $M$. Let $T_g $ be the set of all $\alpha\in\bbF_q$ such that $g(F(\alpha))=0$ or $f(\alpha)=0$. Therefore, $$|T_g|\cdot M\leq \deg (Q)\leq \nicefrac{r}{d}\cdot qM+4qm\implies |T_g|\leq \frac{r}d+4q\frac{m}{M}$$So now we choose $M=\lt[\sqrt{\nicefrac{2q}d}\rt]-3$. Since $q>100dm^2$ we have $M\geq \sqrt{\nicefrac{2q}d}-4\geq \sqrt{\nicefrac{q}d}\geq m+1$. Therefore, we get $$ |T_g|\leq \frac{r}{d}q+4md^{\nicefrac12}q^{\nicefrac12}$$

  Now take $g(X)=X-1$. Then $r-=1$. Observe that in this case $T_g$ is the set of all $\alpha\in\bbF_q$ such that either $F(\alpha)=1$ or $f(\alpha)=0$. Then $|T_g|=Z_0+Z_1$ where $Z_0$ and $Z_1$ are defined in \StepanovSolCountlem. Thus, $$|T_g|=Z_0+Z_1\leq \frac{q}{d}+4md^{\nicefrac12}q^{\nicefrac12}$$Therefore, $$Z(Y^d=f(X))=Z_0+Z_1\leq d\cdot |T_g|\leq q+4md^{\nicefrac32}q^{\nicefrac12}\implies Z(Y^d=f(X))-q\leq 4md^{\nicefrac32}q^{\nicefrac12}$$Now we have to show the lower bound. 

  Now select $g(X)=X^{d-1}+\cdots +X+1$. Then $r=d-1$. Notice that in this case $T_{g}=Z_0+Z_2$. Therefore, in this case we have $$|T_g|=Z_0+Z_2\leq \frac{d-1}{d}q+4md^{\nicefrac12}q^{\nicefrac12}\implies Z_1=q-Z_0-Z_2\geq \frac{q}d-4md^{\nicefrac12}q^{\nicefrac12}$$Therefore we have $$Z(Y^d=f(X))\geq d\cdot Z_1\geq q-4md^{\nicefrac32}q^{\nicefrac12}$$Thus, we have the theorem. 
\end{proof}
\subsubsection{Removal of the assumption \texorpdfstring{$\gcd(m,d)=1$}{gcd(m, d)=1}}
In this part we will remove the assumption that $\gcd(m,d)=1$. So now we can not draw conclusion from $\gcd(d,m)=1$ that $Y^d-f(X)$ is absolutely irreducible. We have to prove \StepanovRestrictedthm just using the assumption that $Y^d-f(X)$ is absolutely irreducible. 

\nt{Now observe that this condition was only required in the proof of \StepanovLinIndeplem. 
}

\paragraph*{Reduction to $f(0)\neq 0$.} We can assume $f(0)\neq 0$ without loss of generality. Since $f$ is a non-zero polynomial there exists $c\in\bbF_q$ such that $f(c)\neq 0$. Then we can replace $f$ by $f(X+c)$ and the number of solutions of $Y^d=f(X)$ will not change as well as the polynomials $h_i(X+C)$ in \StepanovLinIndeplem and \StepanovQExistslem doesn't change. So from now on we will assume that $f(0)\neq 0$.

\paragraph*{Constructing Auxiliary polynomials.} Let $\zeta_1,\dots, \zeta_d\in\ov{\bbF_q}$ be the roots of $X^d-1$, so $X^d-1=(X-\zeta_1)\cdots (X-\zeta_d)$. Working in $\ov{\bbF_q}[T_0,\dots,T_{d-1},Y]$ with formal variables $T_0,\dots,T_{d-1},Y$, consider the polynomial $$\hH(Y,T_0,\dots, T_{d-1})=T_0+T_1 Y+\cdots+T_{d-1}Y^{d-1}$$ and define $$G=\prod_{i=1}^d \hH(\zeta_iY,T_0,\dots, T_{d-1})\in\ov{\bbF_q}[T_0,\dots,T_{d-1},Y].$$ Since permuting the $\zeta_i$'s only permutes the factors of the product, $G$ is symmetric in $\zeta_1Y,\dots,\zeta_dY$. By \ElmSymmetrySpansthm, $G$ is a polynomial in elementary symmetric polynomials in $\zeta_1Y,\dots, \zeta_dY$. 
Since $X^d-1=(X-\zeta_1)\cdots (X-\zeta_d)$ we have $$\esym_j(\zeta_1,\dots,\zeta_d)\equiv 0\text{ for all } j\in[d-1],\quad\text{and}\quad \esym_d(\zeta_1,\dots,\zeta_d)\equiv =-Y^{d+1}.$$ Therefore, every monomial of $G$ with non-zero $Y$-degree as $Y$-degree multiples of $d$. Hence, there exists $\hG\in\ov{\bbF_q}[T_0,\dots,T_{d-1},Y]$ such that $G\equiv \hG(Y^d,T_0,\dots,T_{d-1})$ where $\deg_Y(\hG)=d-1$ and for all $j\in\{0,\dots,d-1\}$, $\deg_{T_j}(\hG)=d$. So now we construct the polynomial $P\in\bbF_q[Y_1,Y_2,T_0,\dots, T_{d-1}]$ as $$P(Y_1,Y_2,T_0,\dots, T_{d-1})=Y_2^{d-1}\cdot \hG\lt(\frac{Y_1}{Y_2},T_0,\dots, T_{d-1}\rt)$$then $\deg(Y_1)P=d-1$ and $\deg_{Y_2}(P)=d-1$ and $\deg_{T_j}=d$ for all $j\in\{0,\dots, d-1\}$. 

\begin{want}
We want to show that \StepanovLinIndeplem holds even with the only assumption that  $Y^d-f(X)$ is absolutely irreducible. After that we can apply \StepanovQExistslem and therefore we have \StepanovRestrictedthm without the assumption $\gcd(m,d)=1$.
\end{want}

\paragraph*{Getting to the Case of $\bmP$ applied on $\bm{h_{i0}}$'s.} Suppose $h_0,\dots, h_{d-1}\in \bbF_q[X]$ with $h_i(X)=\sum_{j=0}^th_{ij}(X)\cdot X^{qj}$ for all $ 0\leq i\leq d-1$ with degree bounds as defined in \StepanovLinIndeplem such that $$h_0(X)+h_1(X)\cdot F(X)+\cdots +h_{d-1}(X)\cdot F(X)^{d-1}\equiv 0$$Therefore, we have $\hH(F,h_0,\dots, h_{d-1})\equiv 0$. Hence, by construction we have $\hG(F^d,h_0,\dots, h_{d-1})\equiv 0$. Now $F(X)=f^{\nicefrac{(q-1)}{d}}(X)$. So $F(X)^d=f^{q-1}(X)=\nicefrac{f^q(X)}{f(X)}$. Therefore, $$P(f^q,f,h_0,\dots, h_{d-1})=f^{d-1}(X)\hG\lt( \frac{f^q(X)}{f(X)},h_0,\dots, h_{d-1} \rt)=f^{(d-1)}\hG(F^d,h_0,\dots, h_{d-1})\equiv 0$$Gathering all terms with no factor of $X^q$ gives 
\begin{equation}\label{eq:stepanov-p-zero}
  P(f(0),f(X),h_{00},\cdots, h_{d-1,0})+X^q\cdot P'(X)\equiv 0
\end{equation}
Now $\deg(P(f(0),f(X),h_{00},\cdots, h_{d-1,0}))\leq (d-1)m+d\lt(\nicefrac{q}d-m\rt)<q$. Hence, $P(f(0),f(X),h_{00},\cdots, h_{d-1,0})\equiv 0$ by \eqref{eq:stepanov-p-zero}. 

\begin{enough}
It suffices to show that $h_{00},\dots, h_{d-1,0}$ are identically zero. If that happens then we can divide every thing by $X^q$ and run the same argument to show that $h_{ij}\equiv 0$ for all $i\in\{0,\dots, d-1\}$ and $j\in\{0,\dots, t\}$. 
\end{enough}

\paragraph*{Concluding $\bm{h_{i,0}\equiv 0}$.} Let $\gm\in\ov{\bbF_q(X)}$ such that $\gm^d=\frac{f(X)}{f(0)}$. Since $Y^d-f(X)$ is absolutely irreducible by \YdFxIrrthm $Y^d-\frac1{f(0)}f(X)$ is also absolutely irreducible. Now we already have $P(f(0),f(X),h_{00},\cdots, h_{d-1,0})\equiv 0$. And $$P(f(0),f(X),h_{00},\cdots, h_{d-1,0})=f^{d-1}(X)\cdot \hG\lt(\frac{f(0)}{f(X)},h_{00},\dots, h_{d-1,0}  \rt)\equiv 0$$Since $f$ is a non-zero polynomial we have $\hG\lt(\frac{f(0)}{f(X)},h_{00},\dots, h_{d-1,0}  \rt)\equiv 0$. Since $\gm^d=\nicefrac{f(X)}{f(0)}$ we have 
\begin{equation}\label{eq:stepanov-g-zero}
  0\equiv \hG\lt(\frac{f(0)}{f(X)},h_{00},\dots, h_{d-1,0}\rt)= \hG\lt(\frac1{\gm^d},h_{00},\dots, h_{d-1,0}\rt)=G\lt(\nicefrac1{\gm},h_{00},\dots, h_{d-1,0}\rt)
\end{equation}
Therefore, there exists $i\in [d]$ such that $\hH\lt(\nicefrac{\zeta_i}{\gm},h_{00},\dots, h_{d-1,0}\rt)\equiv 0$. Expanding it out we obtain $$h_{00}(X)+\frac{\zeta_i}{\gm}h_{10}(X)+\cdots +\lt(\frac{\zeta_i}{\gm}\rt)^{d-1}h_{d-1,0}(X)\equiv 0\iff h_{00}\cdot \gm^{d-1}+h_{10}\cdot \gm^{d-2}\cdot \zeta_i+\cdots +h_{d-1,0}\cdot \zeta_i^{d-1}\equiv 0$$ Let $R(Y)=h_{00}\cdot Y^{d-1}+h_{10}\cdot Y^{d-2}\cdot \zeta_i+\cdots +h_{d-1,0}\cdot \zeta_i^{d-1}\in(X)[Y]$. Now since $Y^d-f(X)$ is absolutely irreducible, $[\ov{\bbF_q}(X,\gm)\colon\ov{\bbF_q}(X)]=d$. But we obtained a polynomial $R(Y)$ of degree $d-1$ over $\ov{\bbF_q}(X)$ such that $R(\gm)\equiv 0$. Therefore, $R(Y)$ is a zero polynomial. So $h_{i,0}\equiv 0$ for all $i\in\{0,\dots, d-1\}$. Hence, we are done. 

So now we have finally proved the \Stepanov.

\subsection{Connection between the Bound and Genus of Curve}

The constant $4md^{\nicefrac32}$ in \Stepanov, while explicit, is not tight as a function of the curve's geometry. For the hyperelliptic case $d=2$, the correct geometric measure of complexity is the \emph{genus} $g$, and H.\ M.\ Stark \cite{Stark_1973_RiemannHyperelliptic} showed that the Stepanov argument can be refined to replace $4md^{\nicefrac32}$ with a constant of size $2g$, recovering the sharp Weil bound.

\paragraph*{Genus of a Hyperelliptic Curve.}
\index{genus!hyperelliptic curve}%
Let $f\in\bbF_q[X]$ have degree $m$ with $m$ distinct roots, so $Y^2-f(X)$ is absolutely irreducible. The smooth projective model $\mathcal{C}$ of the affine curve $Y^2=f(X)$ has genus
\begin{equation}\label{eq:hyperelliptic-genus}
  g = \begin{cases} \dfrac{m-1}{2} & \text{if } m \text{ is odd,} \\[6pt] \dfrac{m-2}{2} & \text{if } m \text{ is even.} \end{cases}
\end{equation}
Equivalently, $g=\lfloor\nicefrac{(m-1)}{2}\rfloor$. The difference between the two cases has a geometric explanation: when $m$ is odd the projective closure of $Y^2=f(X)$ is already smooth at the point at infinity, so the genus is $\frac{m-1}{2}$; when $m$ is even there is a singularity at infinity, and resolving it lowers the genus by $\frac{1}{2}$, giving $\frac{m-2}{2}$. In both cases
$$2g \leq m-1,\quad \text{with equality iff $m$ is odd.}$$
The genus is a birational invariant that captures the arithmetic complexity of $\mathcal{C}$ more faithfully than the raw degree $m$: it counts the number of independent holomorphic differentials on $\mathcal{C}$, and it governs both the structure of the Jacobian and the growth rate of the number of rational points.

\paragraph*{Weil's Theorem in Terms of Genus.}
For any smooth projective absolutely irreducible curve of genus $g$ over $\bbF_q$, Weil's theorem gives
$$\left| N - (q+1) \right| \leq 2g\,q^{\nicefrac12}$$
where $N$ is the number of $\bbF_q$-rational points (including those at infinity). For the hyperelliptic curve $\mathcal{C}\colon Y^2=f(X)$ with $N=Z(Y^2=f(X))$, writing this out using \eqref{eq:hyperelliptic-genus} yields
\begin{equation}\label{eq:weil-genus}
  |N-q| \leq 2g\,q^{\nicefrac12}\ (m \text{ odd}), \qquad |N-q+1| \leq 2g\,q^{\nicefrac12}\ (m \text{ even}).
\end{equation}
The constant $2g$ is sharp: it cannot be replaced by any smaller constant independent of $q$. Comparing with the Stepanov bound $|N-q|<4m\cdot 2^{\nicefrac32}\,q^{\nicefrac12}\approx 11.3\,m\,q^{\nicefrac12}$, the Weil constant $2g\leq m-1$ is smaller by roughly a factor of $5.6$.

\paragraph*{Stark's Sharpening.}
For $q=p$ prime and $f$ with $m$ distinct roots, Stark \cite{Stark_1973_RiemannHyperelliptic} proved the sharper bound
$$|N-q| \leq (m-1)\,q^{\nicefrac12}.$$
Translating via \eqref{eq:hyperelliptic-genus}:
\begin{equation}\label{eq:stark-genus}
  |N-q| \leq 2g\,q^{\nicefrac12}\ (m \text{ odd}),\qquad |N-q| \leq (2g+1)\,q^{\nicefrac12}\ (m \text{ even}).
\end{equation}
For odd $m$ this matches the Weil bound \eqref{eq:weil-genus} exactly. For even $m$ it falls short by $q^{\nicefrac12}$, which is explained by the same infinity-point singularity that reduces the genus by $\frac{1}{2}$.

\subsection{Bound for Multiplicative Character Sum of \texorpdfstring{$f(X)$}{f(X)}}

To bound $\sum_{\alpha\in\bbF_q}\psi(f(\alpha))$ via \Stepanov, we first need to understand when $f(X)$ is a $d$-th power in $\bbF_q[X]$, since that case must be handled separately.
\begin{lemma}{Characterisation of $d$-th Powers in $\bbF_q[X]$}{lem:dth-power-char}
  Let $f(X)\in\bbF_q[X]$ and let $d\mid q-1$. Then the following are equivalent:
  \begin{enumerate}[label=(\roman*)]
    \item $f(X)=ak^d(X)$ with $a\in\bbF_q$ and $k(X)\in\bbF_q[X]$.
    \item $f(X)=h^d(X)$ with $h(X)\in\ov{\bbF_q}[X]$.
    \item $f(X)=a(X-\alpha_1)^{k_1}\cdots (X-\alpha_m)^{k_m}$ with $\alpha_i\in\ov{\bbF_q}$ and $d\mid k_i$ for all $i\in[m]$.
  \end{enumerate}
\end{lemma}
\begin{proof}
  We will prove this in three steps. 
  \paragraph*{(i)$\implies$(ii):} Let $\gm\in\ov{\bbF_q}$ be such that $\gm^d=a$. Then take $h(X)=\gm\cdot k(X)$. Then we have $f(X)=h^d(X)$. 
  \paragraph*{(ii)$\implies $(iii):} This direction comes directly. 
  \paragraph*{(iii)$\implies$(i):} Consider the polynomial $k(X)=\prod_{i=1}^m(X-\alpha)i)^{\nicefrac{k_i}m}$. Then $f(X)=a\cdot k^d(X)$. 
  
  \begin{enough}
    $k(X)\in\bbF_q[X]$. 
  \end{enough}
  Since $a\in\bbF_q$ we have $k^d(X)\in\bbF_q[X]$. Let $k(X)=X^u+c_1X^{u-1}+\cdots +c_u$ where $c_i\in\ov{\bbF_q}$ for all $i\in[u]$. Now we look at the coefficients of $k^d(X)$. We will prove via induction that each $c_i\in\bbF_q$. The coefficient of $X^{du-1}$ is $d\cdot c_1$. As $d\neq 0$ and $d\in\bbF_q$, we have $c_1\in\bbF_q$. So suppose $c_1,\dots, c_{i-1}\in\bbF_q$. Now the coefficient of $X^{du-i}$ is $dc_i+$ some polynomial over $c_1,\dots, c_{i-1}$. Since $c_1,\dots, c_{i-1}\in \bbF_q$ and coefficient of $X^{du-i}$ is in $\bbF_q$ we have $dc_i\in\bbF_q\implies c_i\in \bbF_q$. Therefore, $k(X)\in\bbF_q[X]$. Hence, we have the lemma.
\end{proof}

When $Y^d - f(X)$ is absolutely irreducible, the Stepanov solution-count bound translates directly into a Weil-type bound on the character sum $\sum_\alpha \psi(f(\alpha))$.
\begin{Theorem}{Bound for $\Wpsi{f}$: Absolutely Irreducible Case}{thm:stepanov-psi-irr}
  Suppose $d\mid q-1$ and let $\psi\in\Mq^{(d)}$ be a non-trivial multiplicative character of exponent $d$. Let $f(X)\in\bbF_q[X]$ be a polynomial of degree $m$ such that $Y^d-f(X)$ is absolutely irreducible. Suppose $q>100dm^3$. Then $$\lt|\sum_{\alpha\in\bbF_q}\psi(f(\alpha))\rt|<5md^{\nicefrac{3}{2}}q^{\nicefrac12}.$$
\end{Theorem}
\begin{proof}
  Let $g$ be a primitive element of $\bbF_q$. Let $N_{k}$ be the number of $\alpha\in\bbF_q$ such that $f(\alpha)$ is in the coset $g^k\cdot {\bbF_q^*}^{(d)}$.in the group $\quotient{\bbF_q^*}{{\bbF_q^*}^{(d)}}$. Then we have $$\Wpsi{f}= \sum_{k=0}^{d-1}N_k\cdot \psi(g^k)$$Let $Z_k$ be the number of $(\alpha,\beta)\in\bbF_q^2$ such that $\beta^d=f(\alpha)\cdot g^{-k}$. Since $Y^d-f(X)$ is absolutely irreducible, by \YdFxIrrthm we have $Y^d-f(X)\cdot g^{-k}$ is also absolutely irreducible. So by \Stepanov we get $$|Z_k-q|<4md^{\nicefrac32}q^{\nicefrac12}$$Let $\ovZ_k$ be the number of solutions $(\alpha,\beta)\in\bbF_q^2$ such that $\beta^d=f(\alpha)\cdot g^{-k}$ and $\beta\neq 0$. Then $|Z_k-\ovZ_k|\leq m$. So we have $$|\ovZ_k-q|<5md^{\nicefrac32}q^{\nicefrac12}$$Now notice that $N_k=\nicefrac{\ovZ_k}{d}$. We can write $Z_k=\nicefrac{q}d+R_k$. Therefore, we have $$|R_k|< 5md^{\nicefrac12}q^{\nicefrac12}$$So now $$\Wpsi{f}=\sum_{k=0}^{d-1}N_k\cdot \psi(g^k)=\sum_{k=0}^{d-1}\lt(\frac{q}d+R_k\rt)psi(g^k)=\sum_{k=0}^{d-1}R_k\cdot \psi^k(g)$$So we get the bound on the absolute value of $\Wpsi{f}$ as $$|\Wpsi{f}|\leq \sum_{k=0}^{d-1}|R_k|< 5md^{\nicefrac{3}{2}}q^{\nicefrac12}$$Thus, we have the theorem.
\end{proof}

The absolute irreducibility hypothesis in \StepanovPsiIrrthm can be weakened to the condition that $f$ is not a $d$-th power, which covers all remaining non-trivial cases.
\begin{Theorem}{Bound for $\Wpsi{f}$: Non-$d$-th-Power Case}{thm:stepanov-psi-nondth}
  Suppose $d\mid q-1$ and let $\psi\in\Mq^{(d)}$ be a non-trivial multiplicative character with $\ord(\psi)=d>1$. Let $f(X)\in\bbF_q[X]$ be a polynomial of degree $m$ which is not a $d$-th power. Suppose $q>100dm^3$. Then $$\lt|\sum_{\alpha\in\bbF_q}\psi(f(\alpha))\rt|<5md^{\nicefrac{3}{2}}q^{\nicefrac12}.$$
\end{Theorem}
\begin{proof}
  Let $f(X)$ factors as $f(X)=a(X-\alpha_1)^{k_1}\cdots (X-\alpha_t)^{k_t}$ where $\alpha_i$'s are distinct elements of $\ov{\bbF_q}$. Let $e=\gcd(d,k_1,\dots, k_t)$. Suppose $e$ is a proper divisor of $d$. Then consider the polynomial $$k(X)=\prod_{i=1}^t(X-\alpha_i)^{\nicefrac{k_i}e}$$then by \DthPowerCharlem $f(X)=a\cdot k(X)^e$. Now in $k(X)$, $\gcd(\nicefrac{d}e,\nicefrac{k_1}e,\dots,\nicefrac{k_t}e)=1$. So by \YdFxIrrthm we have $Y^{\nicefrac{d}e}-k(X)$ is absolutely irreducible. Now the character $\psi^e$ is also of exponent $\nicefrac{d}e$. So for any $\alpha\in\bbF_q$ we have $$\psi(f(\alpha))=\psi(a)\cdot \psi^e(k(\alpha))$$So by \StepanovPsiIrrthm we have $$|\Wpsi{f}|=\lt|\sum_{\alpha\in\bbF_q}\psi(a)\cdot \psi^r(k(\alpha)\rt|=\lt|\sum_{\alpha\in\bbF_q}\psi^e(k(\alpha))\rt|<5\lt(\frac{m}e\rt)\lt(\frac{d}e\rt)^{\nicefrac{3}{2}}q^{\nicefrac12}\leq 5md^{\nicefrac{3}{2}}q^{\nicefrac12}$$Thus, we have the theorem.
\end{proof}
