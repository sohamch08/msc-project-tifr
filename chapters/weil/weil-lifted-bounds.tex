\section{Weil Bounds via the Lifting Method}\label{sec:weil-lifted-bounds}

In this section we will try to give bounds on the absolute value of deviation of number of zeros from the average number of zeros. And then we will give upper bounds on the absolute values of $\Wpsil{k}{f}$, $\Wchil{k}{g}$ and $\Wmixl{k}{f}{g}$. But here we will count the solutions or give the bounds on absolute value on lifted characters in a finite extension field $E=\bbF_{q^k}$ of $\bbF_q$.  For any polynomial $P(X,Y)\in\bbF_q[X,Y]$ we denote the number of solutions of $P(X,Y)=0$ in the field $ E$ by $Z_E(P(X,Y)=0)$. 

\subsection{Bound for Multiplicative Character Sums}\label{subsec:mchar-weil-bound}
Let $\psi\in\Mq$ be a non-trivial multiplicative character. And let $\chi_0\in\Xq$ be the trivial additive character. Let $f(X)\in\bbF_q[X]$ is a monic polynomial with precisely $m$ distinct roots. Let $g(X)=0$. Then $W\lt(\psi,\chi_0;f,0\rt)=\Wpsi{f}$ and $W^{(k)}\lt(\psi^{(k)},\chi_0^{(k)};f,0\rt)=\Wpsil{k}{f}$. So by \LiftedLRootscor there exists complex numbers $w_1,\dots, w_{m-1}\in\bbC$ such that \begin{equation}\label{eq:mchar-complex-root-sum}
  \Wpsi{f}=\sum_{\alpha\in\bbF_{q}}\psi(f(\alpha))=-\sum_{j=1}^{m-1}w_j,\qquad\Wpsil{k}{f}=\sum_{\alpha\in E}\psi^{(k)}(f(\alpha))=-\sum_{j=1}^{m-1}w_j^k
\end{equation}
Suppose $\psi\in\Mq^{(d)}$ where $d>1$. Then there are $d$ characters of exponent $d$. For each such non-trivial character $\psi$ we define complex numbers $w_{\psi,j}\in\bbC$ for all $j\in[m-1]$ such that  $$\Wpsi{f}=-\sum_{j=1}^{m-1}w_{\psi,j},\qquad \Wpsil{k}{f}=-\sum_{j=1}^{m-1}w_{\psi,j}^k$$So taking the sum over all non-trivial $\psi\in\Mq^{(d)}$ we get $$\sum_{\psi\in\Mq^{(d)},\psi\neq \psi_0}\Wpsil{k}{f}=-\sum_{\psi\in\Mq^{(d)},\psi\neq \psi_0}\sum_{j=1}^{m-1}w_{\psi,j}^k$$On the other hand if $\psi$ is trivial then we have $W^{(k)}\lt(\psi_0^{(k)};f\rt)=q^k$. Then we have the following lemma
\begin{lemma}{Counting Solutions of $\beta^d=\alpha$ via Additive Characters}{lem:dth-power-count-mult}
  For a given $\alpha\in E$ the number of $\beta\in E$ with $\beta^d=\alpha$ equals $$\sum_{\psi\in\Mq^{(d)}}\psi^{(k)}(\alpha)=\sum_{\psi\in\Mq^{(d)}}\psi(N_{E/\bbF_q}(\alpha)).$$
\end{lemma}
\begin{proof}
  Since $\alpha\to N_{E/\bbF_q}(\alpha)$ is a group homomorphism from $ E^*\to\bbF_q^*$ which is an onto homomorphism. Now for any $\gm\in\bbF_q^*$ the number of $\beta\in E^*$ such that $N_{E/\bbF_q}(\beta)=\gm$ is exactly $1+q+\cdots +q^{k-1}=\nicefrac{ E^*}{|\bbF_q^*|}$. Then restriction $\lt.N_{E/\bbF_q}\rt|_{{ E^*}^{(d)}}\colon { E^*}^{(d)}\to {\bbF_q^*}^{(d)}$ is also onto. So for any $\alpha\in E$, $N_{E/\bbF_q}(\alpha)\in {\bbF_q^*}^{(d)}\iff \alpha\in{ E^*}^{(d)}$. 
  
  Now by \thmref{thm:char-power-sum-d} we have $\sum_{\psi\in\Mq^{(d)}}\psi^{(k)}(\alpha)\in \{0,1,d\}$. If $\alpha=0$ then the sum is $1$ and otherwise if $N_{E/\bbF_q}\in {\bbF_q^*}^{(d)}$ then the sum is $d$ and if $N_{E/\bbF_q}\notin {\bbF_q^*}^{(d)}$ then the sum is $0$. 
  \begin{enumerate}[label=(\roman*),itemsep=-1mm]
    \item If $\alpha= 0$ and $\alpha\notin { E^*}^{(d)}$ then there is single solution for $\beta=0$ such that $\beta^d=\alpha$. 
    \item If $\alpha\neq 0$ and $\alpha\notin { E^*}^{(d)}$ then there are no solution for $\beta$ such that $\beta^d=\alpha$. 
    \item If $\alpha\in { E^*}^{(d)}$ then there are $d$ solutions for $\beta\in E$ such that $\beta^d=\alpha$.
  \end{enumerate}
  So the sum in the lemma statement exactly gives the number of solutions. 
\end{proof}
Now we apply the above lemma for the solutions of  $Y^d-f(X)$ and count the number of solutions using multiplicative character sums. So we take the above sum over all possible $\alpha\in E$, and we get the following lemma:
\begin{lemma}{Solutions of $Y^d=f(X)$ via Character Sums}{lem:ydminusfx-charsum}
  Suppose $f(X)\in\bbF_q[X]$ has precisely $m$ distinct roots. Then $$Z_E\lt(Y^d-f(X)\rt)=\sum_{\psi\in\Mq^{(d)}}\sum_{\alpha\in\bbF_q}\psi^{(k)}(f(\alpha))=\sum_{\psi\in\Mq^{(d)}}\Wpsil{k}{f}.$$
\end{lemma}
We prove the Weil bound for polynomials like $Y^d-f(X)$ in three steps below. And we also show bound of the absolute value of the complex numbers $w_{\psi,j}$  for all non-trivial $\psi\in\Mq^{(d)}$ and $j\in[m-1]$. 
\begin{lemma}{Reciprocal Roots of $\Wpsil{k}{f}$ have Modulus $\leq q^{1/2}$}{lem:weil-roots-modulus-psi}
  Let $f(X)\in\bbF_q[X]$ be a polynomial with precisely $m$ distinct roots, and let $\psi\in\Mq$ be a non-trivial multiplicative character of exponent $d$. If $Y^d-f(X)$ is absolutely irreducible or $f(X)$ is not a $d^{th}$ power, then for every $j\in[m-1]$ we have $|w_{\psi,j}|\leq q^{\nicefrac12}$.
\end{lemma}
\begin{proof}
  We can assume $f$ to be monic since $\psi(a\cdot f(\alpha))=\psi(a)\cdot \psi(f(\alpha)$ and hence $\psi$ evaluated on the leading coefficient gets out of the sum which doesn't change the absolute value. So we will assume $f$ is monic. Now by \Stepanov, $|Z_E(Y^d=f(X))-q^k|=O(q^{\nicefrac{k}2})$. Since $W^{(k)}\lt(\psi_0^{(k)};f\rt)=q^k$, \YdFxCharSumlem gives $$\lt|\sum_{\psi\in\Mq^{(d)},\,\psi\neq\psi_0}\Wpsil{k}{f}\rt|=O(q^{\nicefrac{k}2}).$$Therefore, $|\Wpsil{k}{f}|=O(q^{\nicefrac{k}2})$ for each non-trivial $\psi$. Since $\Wpsil{k}{f}=-\sum_{j=1}^{m-1}w_{\psi,j}^k$, \PowerSumBound gives $|w_{\psi,j}|\leq q^{\nicefrac12}$ for all $j$.
\end{proof}

Now by \lemref{lem:weil-roots-modulus-psi}, $|w_{\psi,j}|\leq q^{\nicefrac12}$ for all $j\in[m-1]$. Hence, $|\Wpsi{f}|\leq (m-1)q^{\nicefrac12}$ and $|\Wpsil{k}{f}|\leq (m-1)q^{\nicefrac{k}2}$. So we have the following theorem. 
\begin{Theorem}{Weil Bound for $\Wpsil{k}{f}$: Absolutely Irreducible Case}{thm:weil-bound-psi-irr}
  \index{Weil bound}\index{Weil bound!multiplicative character sum}%
  Let $\psi\in\Mq$ be a non-trivial multiplicative character of exponent $d$. Suppose $f(X)\in\bbF_q[X]$ has precisely $m$ distinct roots and $Y^d-f(X)$ is absolutely irreducible. Then
  $$\lt|\Wpsi{f}\rt|\leq (m-1)q^{\nicefrac12}\quad\text{and}\quad \lt|\Wpsil{k}{f}\rt|\leq (m-1)q^{\nicefrac{k}2}.$$
\end{Theorem}

We can give the same bound for the case where $f$ is not a $d^{th}$ power instead of the absolute irreducibility of $Y^d-f(X)$. 

\begin{Theorem}{Weil Bound for $\Wpsil{k}{f}$: Non-$d$-th-Power Case}{thm:weil-bound-psi-nondth}
  Let $\psi\in\Mq$ be a non-trivial multiplicative character of exponent $d$. Suppose $f(X)\in\bbF_q[X]$ has precisely $m$ distinct roots and $f(X)$ is not a $d$-th power in $\bbF_q[X]$. Then
  $$\lt|\Wpsi{f}\rt|\leq (m-1)q^{\nicefrac12}\quad\text{and}\quad \lt|\Wpsil{k}{f}\rt|\leq (m-1)q^{\nicefrac{k}2}.$$
\end{Theorem}
\begin{proof}
  Let $f(X)=a(X-\alpha_1)^{k_1}\cdots(X-\alpha_m)^{k_m}$ where $\alpha_1,\dots,\alpha_m\in\ov{\bbF_q}$ are the $m$ distinct roots of $f$. Let $e=\gcd(d,k_1,\dots,k_m)$. Since $f$ is not a $d$-th power in $\bbF_q[X]$, by \DthPowerCharlem we have $e<d$, i.e., $e$ is a proper divisor of $d$.

  Define the polynomial $h(X)=\prod_{i=1}^m(X-\alpha_i)^{\nicefrac{k_i}{e}}$, so that $f(X)=a\cdot h(X)^e$. In $h(X)$ we have $\gcd\lt(\nicefrac{d}{e},\nicefrac{k_1}{e},\dots,\nicefrac{k_m}{e}\rt)=1$, so by \YdFxIrrthm the polynomial $Y^{\nicefrac{d}{e}}-h(X)$ is absolutely irreducible. The polynomial $h(X)$ has precisely the same $m$ distinct roots $\alpha_1,\dots,\alpha_m$ as $f$.

  Since $f(\alpha)=a\cdot h(\alpha)^e$ for every $\alpha\in\bbF_q$, the character $\psi$ satisfies $\psi^{(k)}(f(\alpha))=\psi^{(k)}(a)\cdot\lt(\psi^{(k)}\rt)^e(h(\alpha)).$
  The character $\lt(\psi^{(k)}\rt)^e$ has exponent $\nicefrac{d}{e}$, and $|\psi(a)|=1$. Applying \WeilBoundPsiIrrthm to $\lt(\psi^{(k)}\rt)^e$ and $h(X)$ (which has $m$ distinct roots and $Y^{\nicefrac{d}{e}}-h(X)$ is absolutely irreducible) gives
  $$\lt|\Wpsil{k}{f}\rt|=\lt|\sum_{\alpha\in\bbF_q}\psi^{(k)}(a)\cdot\lt(\psi^{(k)}\rt)^e(h(\alpha))\rt|=\lt|W^{(k)}\lt(\lt(\psi^{(k)}\rt)^e;h\rt)\rt|\leq (m-1)q^{\nicefrac12}.$$
\end{proof}

\begin{corolary}{Bound on the Number of Solutions of $Y^d=f(X)$ over $ E$}{cor:zk-yd-fx-bound}
  Let $\psi\in\Mq$ be a non-trivial multiplicative character of exponent $d$. Suppose $f(X)\in\bbF_q[X]$ has precisely $m$ distinct roots and either $Y^d-f(X)$ is absolutely irreducible, or $f(X)$ is not a $d$-th power. Then 
  $$\lt|Z_E\lt(Y^d=f(X)\rt)-q^k\rt|\leq (d-1)(m-1)q^{\nicefrac{k}2}, \qquad \lt|Z\lt(Y^d=f(X)\rt)-q\rt|\leq (d-1)(m-1)q^{\nicefrac{1}2}.$$
\end{corolary}
\begin{proof}
  By \YdFxCharSumlem, $Z_E(Y^d=f(X))=q^k+\sum_{\psi\neq\psi_0}\Wpsil{k}{f}$. Since there are $d-1$ non-trivial characters in $\Mq^{(d)}$ and \lemref{lem:weil-roots-modulus-psi} gives $|\Wpsil{k}{f}|\leq (m-1)q^{\nicefrac{k}2}$ for each, the triangle inequality yields the bound. Similarly, we get $\lt|Z\lt(Y^d=f(X)\rt)-q\rt|\leq (d-1)(m-1)q^{\nicefrac{1}2}$
\end{proof}
\subsection{Bound for Additive Character Sums}\label{subsec:achar-weil-bound}
Let $\chi\in\Xq$ be a non-trivial additive character. Let $\psi_0\in\Mq$ be the trivial multiplicative character. Let $g\in\bbF_q$ be a polynomial of degree $n$. Take $f(X)=1$. Then $W\lt(\psi_0,\chi;1,g\rt)=\Wchi{g}$ and $W^{(k)}\lt(\psi_0^{(k)},\chi^{(k)};1,g\rt)=\Wpsil{k}{f}$. So by \LiftedLRootscor there exists complex numbers $w_1,\dots, w_{n-1}$ such that \begin{equation}\label{eq:achar-complex-root-sum}
  \Wchi{g}=\sum_{\alpha\in\bbF_q}\chi(g(\alpha))=-\sum_{j=1}^{n-1}w_j,\qquad \Wchil{k}{g}=\sum_{\alpha\in E}\chi^{(k)}(g(\alpha))=-\sum_{j=1}^{n-1}w_j^k
\end{equation}There are $q$ additive characters $\chi\in\Xq$ of $\bbF_q$. For each non-trivial character $\chi$ we define complex numbers $w_{\chi,j}\in\bbC$ for all $j\in[n-1]$ such that $$\Wchi{g}=-\sum_{j=1}^{n-1}w_{\chi,j},\qquad \Wchil{k}{g}=-\sum_{j=1}^{n-1}w_{\chi,j}^k$$So if we take the sum over all non-trivial additive character we get $$\sum_{\chi\in\Xq,\chi\neq \chi_0}\Wchil{k}{g}=-\sum_{\chi\in\Xq,\chi\neq \chi_0}\sum_{j=1}^{n-1}w_{\chi,j}^k$$On the other hand if $\chi=\chi_0$ i.e. if $\chi$ is trivial then $\Wchil{k}{g}=q^{k}$. Then we have the following lemma
\begin{lemma}{Counting Solutions of $\beta^q-\beta=\alpha$ via Additive Characters}{lem:frobenius-count-add}
  For a given $\alpha\in E$ the number of $\beta\in E$ with $\beta^q-\beta=\alpha$ equals
  $$\sum_{\chi\in\Xq}\chi^{(k)}(\alpha)=\sum_{\chi\in\Xq}\chi\!\lt(\tr_{E/\bbF_q}(\alpha)\rt).$$
\end{lemma}
\begin{proof}
  Now we know if $\tr_{E/\bbF_q}(\alpha)=0$ then we have $q$ solutions for $\beta\in E$ such that $\beta^q-\beta=\alpha$ and by \AddCharPropsthm[(i)] we have $$\sum_{\chi\in\Xq}\chi^{(k)}(\alpha)=q$$If $\tr_{E/\bbF_q}(\alpha)\neq 0$ then there is not $\beta\in E$ such that $\beta^q-\beta=\alpha$. And by \AddCharPropsthm[(i)] we get $\sum_{\chi\in\Xq}\chi^{(k)}(\alpha)=0$. So we have the lemma. 
\end{proof}

Now we apply the above lemma for the solutions of $Y^q-Y-g(X)$ and count the number of solutions using multiplicative character sums. So we take the above sum over all possible $\alpha\in\ E$, and we get the following lemma:
\begin{lemma}{Solutions of $\beta^q-\beta=g(\alpha)$ via Character Sums}{lem:frobenius-charsum}
  Let $\chi\in\Xq$ be a non-trivial additive character and $g(X)\in\bbF_q[X]$ of degree $n$. The number of $(\alpha,\beta)\in E^2$ with $\beta^q-\beta=g(\alpha)$ equals $$\sum_{\chi\in\Xq}\sum_{\alpha\in\bbF_q}\chi^{(k)}(g(\alpha))=\sum_{\chi\in\Xq}\Wchil{k}{g}.$$
\end{lemma}

Now we will prove the Weil bound for the polynomial $Y^q-Y-g(X)$ below. We will also show bound on the absolute value of the complex numbers $w_{\chi,j}$ for all non-trivial $\chi\in \Xq$ and $j\in[n-1]$. 

\begin{lemma}{Reciprocal Roots of $\Wchil{k}{g}$ have Modulus $\leq q^{1/2}$}{lem:weil-roots-modulus-chi}
  Let $\chi\in\Xq$ be a non-trivial additive character and $g(X)\in\bbF_q[X]$ of degree $n$. Suppose that
  \begin{enumerate}[label=(\roman*)]
    \item either $n<q$ and $\gcd(n,q)=1$,
    \item or $Y^q-Y-g(X)$ is absolutely irreducible.
  \end{enumerate}
  Then for every $j\in[n-1]$ we have $|w_{\chi,j}|\leq q^{\nicefrac12}$.
\end{lemma}
\begin{proof}
  We can assume $g(X)$ has constant term to be $0$ since $\chi(g(\alpha)+a)=\chi(g(\alpha))\cdot chi(a)$ and so $\chi$ evaluated on the constant term gets out of the sum, and it doesn't change the absolute value. So we will assume $g$ has constant term to be $0$. Now suppose we have \begin{equation}\label{eq:yq-y-g-sol-bound}
    \lt|Z(Y^q-y-g(X))-q^k\rt|=O(q^{\nicefrac{k}2})
  \end{equation} Since $W^{(k)}\lt(\chi_0^{(k)};g\rt)=q^k$, by \lemref{lem:frobenius-charsum} we get $$\lt|\sum_{\chi\in\Xq,\chi\neq\chi_0}\Wchil{k}{g}\rt|=O(q^{\nicefrac{k}2})$$So \PowerSumBound gives $|w_{\chi,j}|\leq q^{\nicefrac12}$. 
  \begin{enumerate}[label=(\roman*)]
    \item By \Bombieri we have the bound in \eqref{eq:yq-y-g-sol-bound}. So the lemma is true for this case. 
    \item By \Weil the bound in \eqref{eq:yq-y-g-sol-bound} is true. So the lemma holds. 
  \end{enumerate}
  Therefore, we have the lemma. 
\end{proof}
Now by the above lemma we have $|w_{\chi,j}|\leq q^{\nicefrac12}$ for all non-trivial $\chi\Xq$ and for all $j\in[n-1]$. Therefore $|\Wchi{g}|\leq (n-1)q^{\nicefrac12}$ and $\lt|\Wchil{k}{g}\rt|\leq (n-1)q^{\nicefrac{k}2}$. So we have the following theorem.
\begin{Theorem}{Weil Bound for $\Wchil{k}{g}$}{thm:weil-bound-chi}
  \index{Weil bound!additive character sum}%
  Let $\chi\in\Xq$ be a non-trivial additive character and $g(X)\in\bbF_q[X]$ of degree $n$. Suppose that
  \begin{enumerate}[label=(\roman*)]
    \item either $n<q$ and $\gcd(n,q)=1$,
    \item or $Y^q-Y-g(X)$ is absolutely irreducible.
  \end{enumerate}
  Then $$\lt|\Wchi{g}\rt|\leq (n-1)q^{\nicefrac12}\quad\text{and}\quad \lt|\Wchil{k}{g}\rt|\leq (n-1)q^{\nicefrac{k}2}.$$
\end{Theorem}
\begin{corolary}{Bound on the Number of Solutions of $Y^q-Y=g(X)$ over $E$}{cor:ze-yq-y-gx-bound}
  Let $\chi\in\Xq$ be a non-trivial additive character and $g(X)\in\bbF_q[X]$ of degree $n$. Suppose that
  \begin{enumerate}[label=(\roman*)]
    \item either $n<q$ and $\gcd(n,q)=1$,
    \item or $Y^q-Y-g(X)$ is absolutely irreducible.
  \end{enumerate}
  Then $$\lt|Z_E\lt(Y^q-Y=g(X)\rt)-q^k\rt|\leq (q-1)(n-1)q^{\nicefrac{k}2}\qquad \lt|Z\lt(Y^q-Y=g(X)\rt)-q\rt|\leq (q-1)(n-1)q^{\nicefrac{1}2}.$$
\end{corolary}
\begin{proof}
  Now by \WeilRootsModulusChilem we have $Z_E(Y^q-Y=g(X))=q^k+\sum_{\chi\neq \chi_0}\Wchil{k}{g}$. Since there are $q-1$ non-trivial characters in $\Xq$ and \WeilBoundChithm we have $\lt|Z_E(Y^q-Y=g(X))-q^k\rt|\leq (q-1)(d-1)q^{\nicefrac{k}2}$ and $\lt|Z(Y^q-Y=g(X))-q\rt|\leq (q-1)(d-1)q^{\nicefrac12}$. 
\end{proof}
\subsection{Bound for Mixed Character Sums}
Suppose $f,g\in\bbF_q[X]$ such that $f$ is monic and $f$ has precisely  $m$ distinct roots in $\ov{\bbF_q}$ and $g$ has degree $n$ with constant term $0$. Let $\psi\in\Mq^{(d)}$ be non-trivial multiplicative character of exponent $f$ and $\chi\in\Xq$ be a non-trivial additive character of $\bbF_q$. So by \LiftedLRootscor there exists complex numbers $w_1,\dots, w_{n+m-1}$ such that $$\Wmix{f}{g}=\sum_{\alpha\in\bbF_q}\psi(f(\alpha))\chi(g(\alpha))=-\sum_{j=1}^{n+m-1}w_j,\quad\Wmixl{k}{f}{g}=\sum_{\alpha\in E}\psi^{(k)}(f(\alpha))\cdot \chi^{(k)}(g(\alpha))=-\sum_{j=1}^{n+m-1}w_j^k$$Now for every non-trivial multiplicative character $\psi\in\Mq^{(q)}$ or exponent $d$ and every non-trivial additive character $\chi\in Xq$ we define the complex numbers $w_{\psi,\chi,j}\in\bbC$ for all $j\in[n+m-1]$ such that $$\Wmix{f}{g}=-\sum_{j=1}^{n+m-1}w_{\psi,\chi,j},\qquad \Wmixl{k}{f}{g}=-\sum_{j=1}^{n+m-1}w^k_{\psi,\chi,j}$$ On the other hand if at least one of $\psi,\chi$ becomes trivial then we have 
\begin{enumerate}[label=(\roman*),itemsep=-1mm]
  \item If $\psi$ is trivial i.e. $\psi=\psi_0$ but $\chi$ is non-trivial then $W^{(k)}\lt(\psi_0^{(k)},\chi^{(k)};f,g\rt)=\Wchil{k}{g}$ since for all $\alpha$, $\psi_0(f(\alpha))=1$. and so by the equation \eqref{eq:achar-complex-root-sum} in  \autoref{subsec:achar-weil-bound} we have $$W^{(k)}\lt(\psi_0^{(k)},\chi^{(k)};f,g\rt)=\sum_{\alpha\in E}\chi^{(k)}(g(\alpha))=-\sum_{j=1}^{n-1}w_{\chi,j}^k$$ and similarly we have $W\lt(\psi_0,\chi;f,g\rt)=\Wchil{k}{g}=-\sum_{j=1}^{n-1}w_{\chi,j}$.
  \item If $\chi$ is trivial i.e. $\chi=\chi_0$ but $\psi$ is non-trivial then $W^{(k)}\lt(\psi^{(k)},\chi_0^{(k)};f,g\rt)=\Wpsil{k}{f}$ since for all $\alpha$, $\chi_0(g(\alpha))=1$. and so by the equation \eqref{eq:mchar-complex-root-sum} in \autoref{subsec:mchar-weil-bound} we have $$W^{(k)}\lt(\psi^{(k)},\chi_0^{(k)};f,g\rt)=\sum_{\alpha\in E}\psi^{(k)}(f(\alpha))=-\sum_{j=1}^{n-1}w_{\psi,j}^k$$ and similarly we have $W\lt(\psi,\chi_0;f,g\rt)=\Wpsil{k}{f}=-\sum_{j=1}^{n-1}w_{\psi,j}$.
  \item Now if both $\psi,\chi$ are trivial i.e. $\psi=\psi_0$ and $\chi=\chi_0$ then $W^{(k)}\lt(\psi_0^{(k)},\chi_0^{(k)};f,g\rt)=q^k$. 
\end{enumerate}
So now we will count for any $\alpha_1,\alpha_2,\in E$ the number of $(\beta,\gm)\in E^2$ such that both $$\beta^d=\alpha_1,\quad\text{and}\quad \gm^q-\gm=\alpha_2$$ is satisfied
\begin{lemma}{Joint Counting via Multiplicative and Additive Characters}{lem:joint-count-mult-add}
  For $\alpha_1,\alpha_2\in E$ the number of $(\beta,\gm)\in E^2$ with $\beta^d=\alpha_1$ and $\gm^q-\gm=\alpha_2$ equals $$\sum_{\psi\in\Mq^{(d)}}\sum_{\chi\in\Xq}\psi^{(k)}(\alpha_1)\,\chi^{(k)}(\alpha_2).$$
\end{lemma}
\begin{proof}
  Now $$|\{(\beta,\gm)\in E^2\mid \alpha_1=\beta^d,\ \alpha_2=\gm^q-\gm\}|=|\{\beta\in E\mid \alpha_1=\beta^d\}|\cdot |\{\gm\in E\mid \alpha_2=\gm^q-\gm\}|$$So therefore by \DthPowerCountlem and \FrobeniusCountAddlem we have \begin{align*}
    & |\{\beta\in E\mid \alpha_1=\beta^d\}| = \sum_{\psi\in\Mq^{(d)}}\psi^{(k)}(\alpha_1)\\ 
    & |\{\gm\in E\mid \alpha_2=\gm^q-\gm\}| = \sum_{\chi\in\Xq}\chi^{(k)}(\alpha_2)
  \end{align*}Therefore we have $$|\{(\beta,\gm)\in E^2\mid \alpha_1=\beta^d,\ \alpha_2=\gm^q-\gm\}|  = \lt( \sum_{\psi\in\Mq^{(d)}}\psi^{(k)}(\alpha_1)\rt)\lt(\sum_{\chi\in\Xq}\chi^{(k)}(\alpha_2)\rt) = \sum_{\psi\in\Mq^{(d)}}\sum_{\chi\in\Xq}\psi^{(k)}(\alpha_1)\chi^{(k)}(\alpha_2)$$
    So we have the lemma. 
\end{proof}

Now we apply the above lemma for the solutions of $Y^d-f(X)$ and $Z^q-Z-g(X)$ and count the number of common solutions using multiplicative and additive character sums. So we take the sum over all possible $\alpha\in E$ for $f$ and $g$ and we get the following lemma:
\begin{lemma}{Solutions of $\beta^d=f(\alpha)$, $\gamma^q-\gamma=g(\alpha)$ via Character Sums}{lem:joint-solutions-charsum}
  The number of $(\alpha,\beta,\gm)\in E^3$ with $\beta^d=f(\alpha)$ and $\gm^q-\gm=g(\alpha)$ is $$\sum_{\psi\in\Mq^{(d)}}\sum_{\chi\in\Xq}\Wmixl{k}{f}{g}$$
\end{lemma}

So now we give a general Weil Bound for the number of common solutions of $Y^d-f(X)$ and $Z^q-Z-g(X)$. We will have to use a very non-trivial theorem \thmref{thm:absolute-variety-sols} from algebraic geometry. We will assume that theorem. And then give a bound on the absolute value of mixed weil sum.
\begin{Theorem}{Weil Bound for Mixed Character Sums}{thm:weil-bound-mixed}
  \index{Weil bound!mixed character sum}%
  Let $\psi\in\Mq$ and $\chi\in\Xq$ be non-trivial multiplicative and additive characters with $\ord(\psi)=d$, $d\mid q-1$. Let $f(X)\in\bbF_q[X]$ have precisely $m$ distinct roots and $g(X)\in\bbF_q[X]$ with $\deg(g)=n$. Suppose that
  \begin{enumerate}[label=(\roman*)]
    \item either $\gcd(d,\deg(f))=\gcd(n,q)=1$,
    \item or both $Y^d-f(X)$ and $Z^q-Z-g(X)$ are absolutely irreducible.
  \end{enumerate}
  Then $$\lt|\sum_{\alpha\in\bbF_q}\psi(f(\alpha))\chi(g(\alpha))\rt|\leq (m+n-1)q^{\nicefrac{1}{2}}.$$
\end{Theorem}
\begin{proof}
  Like in the previous two sections we can assume $f$ to be monic and $g$ has constant term zero since $\psi(af(\alpha))=\psi(a)\cdot \psi(f(\alpha))$ and $\chi(b+g(\alpha))=\chi(b)\cdot \chi(g(\alpha))$ and so $\psi$, $\chi$ evaluated $a$ and $b$ respectively gets out of the sum and it doesn't change the absolute value. So we will assume $f$ monic and $g$ has constant term zero. Now suppose we already have $$\lt|Z_E\lt(Y^d=f(X), Z^q-Z=g(X)\rt)-q^k\rt|=O(q^{\sqrt{k}2})$$We have  $W^{(k)}\lt(\psi_0^{(k)},\chi_0^{(k)};f,g\rt)=q^k$ by \JointSolCharSumlem we have $$\lt|\sum_{\psi\in\Mq^{(d)},\psi\neq \psi_0}\sum_{\chi\in\Xq,\chi\neq \chi_0}\Wmixl{k}{f}{g}+\sum_{\psi\in\Mq^{(d)},\psi\neq \psi_0}\Wpsil{k}{f}+\sum_{\chi\in\Xq,\chi\neq \chi_0}\Wchil{k}{g} \rt|=O(q^{\nicefrac{k}{2}})$$Now by \WeilRootsModulusChilem and \WeilRootsModuluslem, $|w_{\psi,i}|\leq q^{\nicefrac12}$, $|w_{\chi,j}|\leq q^{\nicefrac12}$ for all $i\in[m-1]$, $\psi\in\Mq^{(d)}$ where $\psi\neq \psi_0$ and $j\in[n-1]$, $\chi\in\Xq$ where $\chi\neq \chi_0$. Therefore by \PowerSumBound we have $|w_{\psi,\chi,j}|\leq q^{\nicefrac{1}{2}}$ for all $j\in[n+m-1]$ and $\psi\in\Mq^{(d)}$ with $\psi\neq \psi_0$ and $\chi\in\Xq$ with $\chi\neq \chi_0$. Therefore we get $\lt|\Wmixl{k}{f}{g}\rt|\leq (n+m-1)q^{\nicefrac{k}2}$ and $|\Wmix{f}{g}|\leq (n+m-1)q^{\nicefrac12}$. 

  Now all we need to prove $\lt|Z_E\lt(Y^d=f(X), Z^q-Z=g(X)\rt)-q^k\rt|=O(q^{\sqrt{k}2})$.
  Now under the conditions of the theorem $Y^d=f(X)$ and $Z^q-Z=g(X)$ defines a special curve called absolute variety\index{absolute variety}. So by a theorem from algebraic geometry \thmref{thm:absolute-variety-sols} we get the bound. 
\end{proof}
The following theorem helps us to count number of solutions in something called absolute variety in algebraic geometry. The proof of this theorem is beyond our scope so we ommit the proof of the theorem. 
\begin{Theorem}{}{thm:absolute-variety-sols}
  Let $\mcA$ be an absolute variety of dimension $d$ defined over $\bbF_q$. Let $Z_k(\mcA)$ be the number of points $(\alpha_1,\dots, \alpha_n)\in\mcA$ with each coordinate in $\bbF_{q^k}$. Then as $k\to \infty$ we have $$Z_k(\mcA)=q^{kd}+O(q^{k(d-\nicefrac12)})$$
\end{Theorem}

\subsection{General Weil Bounds}
Andr\'{e} Weil (1940) generalized the above bounds for any absolutely irreducible polynomial over a finite field and showed that the deviation of number of solutions from average number of solution in any finite extended field is in the order of square root of the size of the extended field. This estimate follows from the Riemann Hypothesis for curves over finite fields.
\begin{Theorem}{Weil Bound Theorem}{thm:weil}
  Let $f(X,Y)\in\bbF_q[X,Y]$ is an absolutely irreducible polynomial and of total degree $d>0$. If $q>250d^5$ then $$|Z(f(X,Y)=0)-q|<\sqrt{2}d^{\nicefrac52}q^{\nicefrac12}$$
\end{Theorem}
The proof of this theorem use non-trivial amount of algebraic geometry and is covered in the Chapter 3 of \cite{Schmidt1976}. 

In fact Riemann Hypothesis gives a stronger estimate $$|Z(f(X,Y)=0)-q|\leq (d-1)(d-2)q^{\nicefrac12}+c(d)$$ for some constant $c(d)$ which depends on $d=\deg(f)$.
