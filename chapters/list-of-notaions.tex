\multicolumn{2}{@{}l}{\textit{General}}\\[2pt]
$\bbN$ & Set of all natural numbers i.e.,\ positive integers\\
$\bbZ$ & Set of all integers\\
$\bbZ_{0}$ & Set of all non-negative integers\\
$\bbR$ & Set of real numbers\\
$\bbC$ & Set of complex numbers\\
$[n]$ & The set $\{1,2,\dots,n\}$ for $n\in\bbN$\\
$i$ & Imaginary unit: $i^2=-1$\\
$S^1$ & $\{z\in\bbC:|z|=1\}$, the multiplicative group of unit complex numbers\\
$\zeta_m$ & A primitive $m$-th root of unity\\
$[x]$ & Floor function: largest integer $\leq x$; also written $\lfloor x\rfloor$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Finite Fields}}\\[2pt]
$\bbF_q$ & Finite field with $q$ elements\\
$\bbF_q^*$ & Multiplicative group of nonzero elements of $\bbF_q$\\
$\bbF_p$ & Prime field with $p$ elements, $p=\Char(\bbF_q)$\\
${\bbF_q^*}^{(d)}$ & Subgroup of $d$-th powers in $\bbF_q^*$\\
$E,F,L,K$ & Generic notation for finite fields or finite extensions of $\bbF_q$\\
$[E:\bbF_q]$ & Degree of the extension $E/\bbF_q$\\
$\bbF_q[X_1,\dots,X_n]$ & Polynomial ring in $n$ variables over $\bbF_q$\\
$\bbF_q^{\leq d}[X_1,\dots,X_n]$ & Polynomials in $\bbF_q[X_1,\dots,X_n]$ of total degree at most $d$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Characters of Finite Abelian Groups}}\\[2pt]
$G$ & A finite Abelian group with identity element $1_G$\\
$Z_k$ & Cyclic multiplicative group of order $k$\\
$\chi_0$ & Trivial character of $G$: $\chi_0(g)=1$ for all $g\in G$\\
$\ov{\chi}$ & Conjugate (and inverse) character of $\chi$: $\ov{\chi}(g)=\overline{\chi(g)}$\\
$\Hom(A,B)$ & Set of all group homomorphisms from $A$ to $B$\\
$\hG=\Hom(G,S^1)$ & Dual group of $G$; the group of all characters of $G$\\
$\Ann(\chi)$ & Annihilator of character $\chi$: $\{g\in G:\chi(g)=1\}$, a subgroup of $G$\\
$\Ann(H)$ & Annihilator of subgroup $H\leq G$: $\{\chi\in\hG:\chi(h)=1\ \forall h\in H\}$\\
$\ev_g$ & Evaluation character of $\hG$ defined by $\ev_g(\chi)=\chi(g)$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Maps on Finite Fields}}\\[2pt]
$\tr\colon \bbF_q\to\bbF_p$ & Absolute trace function\\
$N\colon \bbF_q\to\bbF_p$ & Absolute norm function\\
$\tr_{E/\bbF_q}$ & Trace function from $E$ to $\bbF_q$\\
$\norm_{E/\bbF_q}$ & Norm function from $E$ to $\bbF_q$\\
$\sg\colon \bbF_q\to\bbF_q$ & Frobenius automorphism: $\sg(\alpha)=\alpha^p$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Additive Characters of $\bbF_q$}}\\[2pt]
$\chi,\tau$ & Generic notation for additive characters\\
$\chi_1$ & Canonical additive character of $\bbF_q$: $\chi_1(\alpha)=e^{2\pi i\,\tr(\alpha)/p}$\\
$\chi_\beta$ & Additive character defined by $\chi_\beta(\alpha)=\chi_1(\beta\alpha)$, for $\beta\in\bbF_q$\\
$\Xq$ & Set of all additive characters of $\bbF_q$\\
$\chi^{(r)}$ & Additive character of $\bbF_q$ lifted to $E$ via $\chi^{(r)}=\chi\circ\tr_{E/\bbF_q}$, $[E:\bbF_q]=r$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Multiplicative Characters of $\bbF_q$}}\\[2pt]
$\psi,\lm$ & Generic notation for multiplicative characters\\
$\psi_0$ & Trivial multiplicative character of $\bbF_q$\\
$\eta$ & Quadratic character of $\bbF_q$ (for odd $q$): $\eta(\alpha)=\pm 1$ according as $\alpha$ is a square\\
$\Mq$ & Set of all multiplicative characters of $\bbF_q^*$\\
$\Mq^{(e)}$ & Set of multiplicative characters of $\bbF_q$ of exponent $e$\\
$\ord(\psi)$ & Order of $\psi$: smallest positive integer $d$ with $\psi^d=\psi_0$\\
$\Leg{\alpha}{q}$ & Legendre symbol; equals $\eta(\alpha)$ for $\alpha\in\bbF_q$\\
$\psi^{(r)}$ & Multiplicative character of $\bbF_q$ lifted to $E$ via $\psi^{(r)}=\psi\circ\norm_{E/\bbF_q}$, $[E\colon \bbF_q]=r$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Gaussian and Jacobi Sums}}\\[2pt]
$G(\psi,\chi)$ & Gaussian sum: $\sum_{\alpha\in\bbF_q^*}\psi(\alpha)\,\chi(\alpha)$, for $\psi\in\Mq$, $\chi\in\Xq$\\[12pt]
$J_\alpha(\lm_1,\dots,\lm_k)$ & Generalized Jacobi sum: $\displaystyle\sum_{\substack{\alpha_1,\dots,\alpha_k\in\bbF_q\\\alpha_1+\cdots+\alpha_k=\alpha}}\lm_1(\alpha_1)\cdots\lm_k(\alpha_k)$\\[16pt]
$J(\lm_1,\dots,\lm_k)$ & Jacobi sum; special case $\alpha=1$ of $J_\alpha$\\
$J_0(\lm_1,\dots,\lm_k)$ & Jacobi sum at zero; special case $\alpha=0$ of $J_\alpha$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Kloosterman Sums}}\\[2pt]
$K(\chi;a,b)$ & Kloosterman sum: $\sum_{\gm\in\bbF_q^*}\chi(a\gm+b\gm^{-1})$, for $\chi\in\Xq$, $a,b\in\bbF_q$\\[10pt]
$K(\chi^{(r)};a,b)$ & Kloosterman sum lifted to the degree-$r$ extension of $\bbF_q$\\
$w_1,w_2$ & Reciprocal roots of the $L$-function associated to a Kloosterman sum\\[6pt]

\multicolumn{2}{@{}l}{\textit{Sali\'{e} and Jacobsthal Sums}}\\[2pt]
$S(\chi,\tau)$ & Sali\'{e} sum: $\sum_{\gm\in\bbF_q^*}\eta(\gm)\,\chi(\gm)\,\tau(\gm^{-1})$, for $\chi,\tau\in\Xq$, $q$ odd\\[10pt]
$H_n(\alpha)$ & Jacobsthal sum: $\sum_{\gm\in\bbF_q}\eta(\gm^{n+1}+\alpha\gm)$, for $\alpha\in\bbF_q^*$\\[10pt]
$I_n(\alpha)$ & Associated Jacobsthal sum: $\sum_{\gm\in\bbF_q}\eta(\gm^n+\alpha)$, for $\alpha\in\bbF_q^*$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Classical Riemann Zeta Function}}\\[2pt]
$\zeta(s)$ & Riemann zeta function: $\sum_{n=1}^\infty n^{-s}$, for $\Re(s)>1$\\[10pt]
$\xi(s)$ & Completed zeta function: $\tfrac12s(s-1)\,\Gm(s/2)\,\pi^{-s/2}\,\zeta(s)$\\
$\Gm(s)$ & Gamma function: $\int_0^\infty e^{-t}t^{s-1}\,dt$\\
$\Lambda(n)$ & Von Mangoldt function: $\log p$ if $n=p^k$, else $0$\\
$\rho$ & Non-trivial zero of $\xi(s)$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Dirichlet $L$-functions over $\bbF_q[X]$}}\\[2pt]
$\Phi$ & Set of all monic polynomials in $\bbF_q[X]$\\
$\Phi_d$ & Monic polynomials in $\bbF_q[X]$ of degree exactly $d$\\
$\irr(\Phi)$ & Set of irreducible monic polynomials in $\Phi$\\
$\mfn(f)=q^{\deg(f)}$ & Norm of the polynomial $f\in\bbF_q[X]$\\
$G$ & Group of rational functions $r=h_1/h_2$ in $\bbF_q(X)$ with $h_1,h_2\in\Phi$ monic\\
$\ovG$ & Subgroup of $G$ on which a given character is defined; extended to $G$ by zero\\
$L(s,\chi)$ & Dirichlet $L$-function: $\sum_{h\in\Phi}\chi(h)\,\mfn(h)^{-s}$, for a multiplicative character $\chi$\\
$\ovL(z,\chi)$ & Power series satisfying $\ovL(z,\chi)=\sum_{h\in\Phi}\chi(h)\,z^{\deg(h)}$ and $\ovL(q^{-s},\chi)=L(s,\chi)$\\[10pt]
$L_k$ & $k$-th log-derivative coefficient: $\displaystyle\sum_{\substack{h\in\irr(\Phi)\\\deg(h)\mid k}}\deg(h)\,\chi(h)^{k/\deg(h)}$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Equations over $\bbF_q$ --- Solution Sets}}\\[2pt]
$\mcV(f)$ & Solution set of $f(X_1,\dots,X_n)=0$ in $\bbF_q^n$\\
$Z(f)$ & Number of solutions: $|\mcV(f)|$\\
$\ov{\mcV}(f)$ & Set of \emph{non-trivial} solutions (excluding $(0,\dots,0)$ if it's a zero) of $f=0$\\
$\ov{Z}(f)$ & Number of non-trivial solutions: $|\ov{\mcV}(f)|$\\
$\mcV(f_1,\dots,f_k)$ & Common solution set: $\mcV(f_1)\cap\cdots\cap\mcV(f_k)$\\
$Z(f_1,\dots,f_k)$ & Count of common solutions: $|\mcV(f_1,\dots,f_k)|$\\
$Z_E(P=0)$ & Number of solutions of $P(X_1,\dots, X_n)=0$ in $E$ where $P\in\bbF_q[X_1,\dots, X_n]$, $[E\colon \bbF_q]>1$\\
$\Om_d$ & Space $\bbF_q^{\leq d}[X_1,\dots,X_n]$ of polynomials of total degree at most $d$\\
$\om(d)$ & Set of $n$-tuples $(i_1,\dots,i_n)\in\bbZ_{0}^n$ with $i_1+\cdots+i_n\leq d$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Quadratic Forms}}\\[2pt]
$C_f$ & Coefficient matrix of a quadratic form $f$: $(C_f)_{ij}=a_{ij}$ where $f=X^\top C_f X$\\
$\det(f)$ & Determinant of the coefficient matrix $C_f$\\
$\vth:\bbF_q\to\bbZ$ & Auxiliary function: $\vth(0)=q-1$ and $\vth(\alpha)=-1$ for $\alpha\neq 0$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Diagonal Equations}}\\[2pt]
$d_i=\gcd(k_i,q-1)$ & Exponent reduced modulo the order of $\bbF_q^*$, for $a_iX_i^{k_i}$ in a diagonal equation\\
$M(d_1,\dots,d_n)$ & $\#\bigl\{(t_1,\dots,t_n)\in\prod_{i}[d_i-1] : \tfrac{t_1}{d_1}+\cdots+\tfrac{t_n}{d_n}\in\bbZ\bigr\}$\\[10pt]
$D=\lcm(d_1,\dots,d_n)$ & LCM of exponents; appears in the closed-form formula for $M(d_1,\dots,d_n)$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Character Sums with Polynomials (Weil Sums)}}\\[2pt]
$W(\chi;f)$ & Additive Weil sum: $\sum_{\alpha\in\bbF_q}\chi(f(\alpha))$, for $\chi\in\Xq$, $f\in\bbF_q[X]$\\[10pt]
$W(\psi;f)$ & Multiplicative Weil sum: $\sum_{\alpha\in\bbF_q}\psi(f(\alpha))$, for $\psi\in\Mq$, $f\in\bbF_q[X]$\\[10pt]
$W(\psi,\chi;f,g)$ & Mixed Weil sum: $\sum_{\alpha\in\bbF_q}\psi(f(\alpha))\,\chi(g(\alpha))$, for $\psi\in\Mq$, $\chi\in\Xq$, $f,g\in\bbF_q[X]$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Lifted Weil Sums (over Extension Fields)}}\\[2pt]
$E=\bbF_{q^k}$ & Degree-$k$ extension of $\bbF_q$; $[E:\bbF_q]=k$\\
$W^{(k)}(\psi^{(k)};f)$ & Lifted multiplicative Weil sum: $\sum_{\alpha\in E}\psi^{(k)}(f(\alpha))$\\[10pt]
$W^{(k)}(\chi^{(k)};g)$ & Lifted additive Weil sum: $\sum_{\alpha\in E}\chi^{(k)}(g(\alpha))$\\[10pt]
$W^{(k)}(\psi^{(k)},\chi^{(k)};f,g)$ & Lifted mixed Weil sum: $\sum_{\alpha\in E}\psi^{(k)}(f(\alpha))\,\chi^{(k)}(g(\alpha))$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Special $L$-functions (Weil chapter)}}\\[2pt]
$\ovG$ (Weil context) & The specific subgroup $\{r\in G : h_1(\gm_i)\cdot h_2(\gm_i)\neq 0\ \forall i\}$ for roots $\gm_i$ of a fixed $f$\\
$\vsg:\ovG\to\bbF_q$ & Multiplicative map $\vsg(r)=\prod_i f(\alpha_i)\cdot\prod_j f(\beta_j)^{-1}$ for $r=\prod(X-\alpha_i)/\prod(X-\beta_j)$\\
$\vrh:G\to\bbF_q$ & Additive map $\vrh(r)=\sum_i g(\alpha_i)-\sum_j g(\beta_j)$ for a fixed polynomial $g$\\
$\xi:G\to\bbC$ & Combined character: $\xi(r)=(\psi\circ\vsg)(r)\cdot(\chi\circ\vrh)(r)$\\
$L(s,\xi)$ & $L$-function on $\xi$: $\sum_{h\in\Phi}\xi(h)\,\mfn(h)^{-s}$\\[10pt]
$\ovL(z,\xi)$ & Power series: $\sum_{h\in\Phi}\xi(h)\,z^{\deg(h)}$, related by $\ovL(q^{-s},\xi)=L(s,\xi)$\\[10pt]
$L^{(k)}(s,\xi^{(k)})$ & $L$-function on the lifted character $\xi^{(k)}$ over $\bbF_{q^k}$\\
$\ovL^{(k)}(z,\xi^{(k)})$ & Power series form of the lifted $L$-function\\
$\Phi^{(k)}$ & Set of monic polynomials over $E=\bbF_{q^k}$\\
$\mfn^{(k)}(h)=q^{k\cdot\deg(h)}$ & Norm of $h\in\Phi^{(k)}$ over $E$; analog of $\mfn(h)=q^{\deg(h)}$ for $\bbF_q$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Absolute Varieties}}\\[2pt]
$\mcA$ & Variety  defined over $\bbF_q$\\
$Z_k(\mcA)$ & Number of $\bbF_{q^k}$-rational points of $\mcA$: $|\{(\alpha_1,\dots,\alpha_n)\in\mcA : \alpha_i\in\bbF_{q^k}\}|$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Hasse Derivatives}}\\[2pt]
$F^{(\ell)}(X)$ & $\ell$-th Hasse derivative of $F(X)$: coefficient of $Z^\ell$ in $F(X+Z)$\\
$\mult(F, \alpha)$ & Multiplicity of vanishing of $F$ at $\alpha$: largest $r$ with $(X-\alpha)^r \mid F(X)$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Pseudopolynomials}}\\[2pt]
$\Lambda(X)$ & The polynomial $X^q - X$ over $\bbF_q$\\
$F_{\langle\ell\rangle}(X)$ & $\ell$-th pseudoderivative of $F$: unique polynomial of degree $<q$ agreeing with $F^{(\ell)}$ on $\bbF_q$\\
$\pdeg(F)$ & Pseudodegree of $F$: $\max_{\ell \geq 0}\deg(F_{\langle\ell\rangle})$\\
$F_i(X)$ & Coefficient polynomials in the base-$\Lambda$ expansion $F = \sum_i F_i \Lambda^i$, with $\deg(F_i) < q$\\[6pt]

\multicolumn{2}{@{}l}{\textit{Decoding from Character Evaluations}}\\[2pt]
$r : \bbF_q \to \{0,\pm 1\}$ & Received (noisy) word; approximates $\psi \circ g$ or $\tr \circ g$\\
$e$ & Number of errors: $|\{\alpha \in \bbF_q : r(\alpha) \neq (\psi \circ g)(\alpha)\}|$\\
$S$ & Error set: $\{\alpha \in \bbF_q : r(\alpha) \neq (\psi \circ g)(\alpha)\}$\\
$Z_S(X)$ & Error locator polynomial: $\prod_{\alpha \in S}(X - \alpha)$\\
$\Delta(r_1, r_2)$ & Hamming distance: $|\{\alpha \in \bbF_q : r_1(\alpha) \neq r_2(\alpha)\}|$
\end{longtable}
